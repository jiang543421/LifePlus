package com.lifepulse.ai.llm.provider;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.lifepulse.ai.llm.LlmClient;
import com.lifepulse.ai.llm.LlmProperties;
import com.lifepulse.ai.llm.LlmRequest;
import com.lifepulse.ai.llm.LlmResponse;
import com.lifepulse.ai.llm.exception.LlmResponseInvalidException;
import com.lifepulse.ai.llm.exception.LlmUnavailableException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpStatusCode;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Component;
import org.springframework.web.client.ResourceAccessException;
import org.springframework.web.client.RestClient;
import org.springframework.web.client.RestClientException;
import java.time.Duration;

/**
 * DeepSeek V3 client (OpenAI-compatible protocol).
 * Conditional on lp.ai.llm.provider=deepseek.
 */
@Component
@ConditionalOnProperty(prefix = "lp.ai.llm", name = "provider", havingValue = "deepseek")
public class DeepSeekClient implements LlmClient {

    private static final Logger log = LoggerFactory.getLogger(DeepSeekClient.class);

    private final RestClient restClient;
    private final LlmProperties props;
    private final ObjectMapper objectMapper;

    public DeepSeekClient(RestClient.Builder builder, LlmProperties props, ObjectMapper objectMapper) {
        this.props = props;
        this.objectMapper = objectMapper;
        this.restClient = builder
            .baseUrl(props.baseUrl())
            .defaultHeader(HttpHeaders.AUTHORIZATION, "Bearer " + props.apiKey())
            .defaultHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
            .build();
    }

    @Override
    public LlmResponse generate(LlmRequest request) {
        long start = System.currentTimeMillis();
        String body = buildBody(request);
        try {
            String responseBody = restClient.post()
                .uri("/chat/completions")
                .body(body)
                .retrieve()
                .onStatus(HttpStatusCode::is4xxClientError, (req, res) -> {
                    int code = res.getStatusCode().value();
                    if (code == 429) {
                        throw new LlmUnavailableException("DeepSeek 429 rate limited");
                    }
                    throw new LlmResponseInvalidException("DeepSeek 4xx: " + code);
                })
                .onStatus(HttpStatusCode::is5xxServerError, (req, res) -> {
                    throw new LlmUnavailableException("DeepSeek 5xx: " + res.getStatusCode().value());
                })
                .body(String.class);

            return parse(responseBody, System.currentTimeMillis() - start);
        } catch (ResourceAccessException ex) {
            // timeout / connection failure
            throw new LlmUnavailableException("DeepSeek network failure: " + ex.getMessage(), ex);
        } catch (LlmUnavailableException | LlmResponseInvalidException ex) {
            throw ex;
        } catch (RestClientException ex) {
            throw new LlmUnavailableException("DeepSeek rest failure: " + ex.getMessage(), ex);
        }
    }

    private String buildBody(LlmRequest req) {
        return String.format("""
            {"model":"%s","max_tokens":%d,"messages":[
              {"role":"system","content":%s},
              {"role":"user","content":%s}
            ]}""",
            props.model(),
            req.maxResponseTokens(),
            jsonString(req.systemPrompt()),
            jsonString(req.userPrompt()));
    }

    private String jsonString(String s) {
        return "\"" + s.replace("\\", "\\\\").replace("\"", "\\\"") + "\"";
    }

    private LlmResponse parse(String body, long latencyMs) {
        if (body == null || body.isBlank()) {
            throw new LlmResponseInvalidException("Empty DeepSeek response body");
        }
        try {
            JsonNode root = objectMapper.readTree(body);
            String content = root.path("choices").path(0).path("message").path("content").asText(null);
            if (content == null) {
                throw new LlmResponseInvalidException("Missing choices[0].message.content");
            }
            int promptTokens = root.path("usage").path("prompt_tokens").asInt(0);
            int completionTokens = root.path("usage").path("completion_tokens").asInt(0);
            return new LlmResponse(content, promptTokens, completionTokens, latencyMs);
        } catch (com.fasterxml.jackson.core.JsonProcessingException ex) {
            throw new LlmResponseInvalidException("DeepSeek JSON parse failed: " + ex.getMessage(), ex);
        }
    }
}
