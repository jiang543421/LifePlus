// frontend/src/__tests__/daily/daily.spec.ts
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { setActivePinia, createPinia } from 'pinia';
import { useDailyStore } from '@/stores/daily';
import { dailyApi } from '@/api/daily';
import type { DailyReportPayload, WeeklyReportPayload } from '@/types/daily';

vi.mock('@/api/daily');

const mkDaily = (): DailyReportPayload => ({
  date: '2026-07-15',
  task: { completedCount: 3, totalCount: 5, completionRate: 0.6,
          statusDistribution: {}, priorityDistribution: {} },
  plan: { eventCount: 2, totalMinutes: 240, categoryDistribution: {}, busiestHour: 14 },
  expense: { totalAmount: '47.00', count: 2, categoryBreakdown: {}, topCategories: [] },
  diet: { enabled: false, value: null, reason: '饮食模块未上线' },
});

describe('useDailyStore', () => {
  beforeEach(() => { setActivePinia(createPinia()); });

  it('fetchDaily populates daily payload', async () => {
    vi.mocked(dailyApi.getDailyReport).mockResolvedValue(mkDaily());
    const s = useDailyStore();
    await s.fetchDaily('2026-07-15');
    expect(s.daily?.task.completionRate).toBe(0.6);
  });

  it('fetchDaily default to today when no date', async () => {
    vi.mocked(dailyApi.getDailyReport).mockResolvedValue(mkDaily());
    const s = useDailyStore();
    await s.fetchDaily();
    expect(dailyApi.getDailyReport).toHaveBeenCalled();
  });

  it('fetchWeekly populates weekly payload', async () => {
    const wk: WeeklyReportPayload = {
      isoWeek: '2026-W29', weekStart: '2026-07-13', weekEnd: '2026-07-19',
      comparison: {
        taskCompletion: { current: 12, previous: 10, delta: 2 },
        planEvents: { current: 8, previous: 5, delta: 3 },
        expenseAmount: { current: 500, previous: 0, delta: null },
      },
    };
    vi.mocked(dailyApi.getWeeklyReport).mockResolvedValue(wk);
    const s = useDailyStore();
    await s.fetchWeekly('2026-W29');
    expect(s.weekly?.isoWeek).toBe('2026-W29');
  });

  it('viewMode toggles between daily and weekly', () => {
    const s = useDailyStore();
    expect(s.viewMode).toBe('daily');
    s.setViewMode('weekly');
    expect(s.viewMode).toBe('weekly');
  });

  it('currentDate advances one day', () => {
    const s = useDailyStore();
    s.setCurrentDate('2026-07-15');
    s.shiftDate(1);
    expect(s.currentDate).toBe('2026-07-16');
  });

  it('shiftDate handles month boundary', () => {
    const s = useDailyStore();
    s.setCurrentDate('2026-07-31');
    s.shiftDate(1);
    expect(s.currentDate).toBe('2026-08-01');
  });

  it('shiftDate handles year boundary', () => {
    const s = useDailyStore();
    s.setCurrentDate('2026-12-31');
    s.shiftDate(1);
    expect(s.currentDate).toBe('2027-01-01');
  });

  it('shiftWeek advances 7 days', () => {
    const s = useDailyStore();
    s.setCurrentDate('2026-07-15');
    s.shiftWeek(1);
    expect(s.currentDate).toBe('2026-07-22');
  });

  it('diet.enabled is always false in v1.2.3', () => {
    const s = useDailyStore();
    expect(s.daily).toBeNull();  // 初始未加载
    // 加载后断言(用 mock)
    s.daily = mkDaily();
    expect(s.daily?.diet.enabled).toBe(false);
  });

  it('1004 error captured in error state', async () => {
    vi.mocked(dailyApi.getDailyReport).mockRejectedValue(
      Object.assign(new Error('validation'), { response: { data: { code: 1004, message: '未来日期' } } }));
    const s = useDailyStore();
    await expect(s.fetchDaily('2099-01-01')).rejects.toThrow();
    expect(s.error?.code).toBe(1004);
  });
});

```typescript
// frontend/src/stores/daily.ts
import { defineStore } from 'pinia';
import dayjs from 'dayjs';
import { dailyApi } from '@/api/daily';
import type { DailyReportPayload, WeeklyReportPayload } from '@/types/daily';

export type ViewMode = 'daily' | 'weekly';
export interface DailyError { code: number; message: string }

export const useDailyStore = defineStore('daily', () => {
  const daily = ref<DailyReportPayload | null>(null);
  const weekly = ref<WeeklyReportPayload | null>(null);
  const currentDate = ref(dayjs().format('YYYY-MM-DD'));
  const currentWeek = ref('');  // 后端默认
  const viewMode = ref<ViewMode>('daily');
  const loading = ref(false);
  const error = ref<DailyError | null>(null);

  async function fetchDaily(date?: string) {
    loading.value = true; error.value = null;
    try {
      const target = date ?? currentDate.value;
      daily.value = await dailyApi.getDailyReport(target);
      currentDate.value = target;
    } catch (e: any) {
      error.value = { code: e.response?.data?.code ?? 500,
                      message: e.response?.data?.message ?? '加载失败' };
      throw e;
    } finally { loading.value = false; }
  }

  async function fetchWeekly(isoWeek?: string) {
    loading.value = true; error.value = null;
    try {
      const target = isoWeek ?? currentWeek.value;
      weekly.value = await dailyApi.getWeeklyReport(target);
      currentWeek.value = target;
    } catch (e: any) {
      error.value = { code: e.response?.data?.code ?? 500,
                      message: e.response?.data?.message ?? '加载失败' };
      throw e;
    } finally { loading.value = false; }
  }

  function setViewMode(mode: ViewMode) { viewMode.value = mode; }

  function setCurrentDate(d: string) { currentDate.value = d; }

  function shiftDate(days: number) {
    currentDate.value = dayjs(currentDate.value).add(days, 'day').format('YYYY-MM-DD');
  }

  function shiftWeek(weeks: number) {
    currentDate.value = dayjs(currentDate.value).add(weeks * 7, 'day').format('YYYY-MM-DD');
  }

  return { daily, weekly, currentDate, currentWeek, viewMode, loading, error,
           fetchDaily, fetchWeekly, setViewMode, setCurrentDate, shiftDate, shiftWeek };
});
```

> **Vue auto-import**:若项目用 `unplugin-auto-import`,`ref` 不需手动 import;否则顶部加 `import { ref } from 'vue'`。
