- [ ] **Step 1**: 创建 `TaskMetricProvider`(基于 `t_task.completed_at`)

```java
package com.lifepulse.daily.provider;

import com.lifepulse.daily.DailyConstants;
import com.lifepulse.daily.TaskMetrics;
import com.lifepulse.task.repository.TaskMapper;
import org.springframework.stereotype.Component;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Map;

@Component
public class TaskMetricProvider implements MetricProvider<TaskMetrics> {

  private final TaskMapper taskMapper;

  public TaskMetricProvider(TaskMapper taskMapper) {
    this.taskMapper = taskMapper;
  }

  @Override
  public TaskMetrics aggregateDaily(long userId, LocalDate date) {
    LocalDateTime from = date.atStartOfDay(DailyConstants.ZONE).toLocalDateTime();
    LocalDateTime to = date.plusDays(1).atStartOfDay(DailyConstants.ZONE).toLocalDateTime().minusNanos(1);
    // 任务完成数 = 截止日在当天 OR 完成日在当天 且 status=DONE
    long completed = taskMapper.countCompletedByUserBetween(userId, from, to);
    long total = taskMapper.countByUserBetween(userId, from, to);
    Map<String, Long> status = taskMapper.statusDistribution(userId, from, to);
    Map<String, Long> priority = taskMapper.priorityDistribution(userId, from, to);
    double rate = total == 0 ? 0.0 : (double) completed / total;
    return new TaskMetrics(completed, total, rate, status, priority);
  }
}
```

> **TaskMapper 新增方法**(若不存在):`countCompletedByUserBetween` / `countByUserBetween` / `statusDistribution` / `priorityDistribution`。
> 若旧版 mapper 用 `dueDate` 字段,沿用既有查询,不强制新建,优先复用。

- [ ] **Step 2**: 创建 `PlanMetricProvider`(基于 `t_plan.start_at`)

```java
package com.lifepulse.daily.provider;

import com.lifepulse.daily.DailyConstants;
import com.lifepulse.daily.PlanMetrics;
import com.lifepulse.plan.repository.PlanMapper;
import org.springframework.stereotype.Component;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Map;

@Component
public class PlanMetricProvider implements MetricProvider<PlanMetrics> {

  private final PlanMapper planMapper;

  public PlanMetricProvider(PlanMapper planMapper) {
    this.planMapper = planMapper;
  }

  @Override
  public PlanMetrics aggregateDaily(long userId, LocalDate date) {
    LocalDateTime from = date.atStartOfDay(DailyConstants.ZONE).toLocalDateTime();
    LocalDateTime to = date.plusDays(1).atStartOfDay(DailyConstants.ZONE).toLocalDateTime().minusNanos(1);
    long count = planMapper.countByUserBetween(userId, from, to);
    long minutes = planMapper.totalMinutesByUserBetween(userId, from, to);
    Map<String, Long> catDist = planMapper.categoryDistribution(userId, from, to);
    Integer busiest = planMapper.busiestHour(userId, from, to);
    return new PlanMetrics(count, minutes, catDist, busiest);
  }
}
```

> **PlanMapper 新增方法**:同上风格,复用既有 SQL 命名约定。

package com.lifepulse.daily.provider;

import com.lifepulse.daily.TaskMetrics;
import com.lifepulse.task.repository.TaskMapper;
import org.junit.jupiter.api.*;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.junit.jupiter.api.extension.ExtendWith;

import java.time.LocalDate;
import java.util.Map;

import static org.assertj.core.api.Assertions.*;
import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class TaskMetricProviderTest {

  @Mock TaskMapper taskMapper;

  TaskMetricProvider provider;

  @BeforeEach
  void setUp() { provider = new TaskMetricProvider(taskMapper); }

  @Test void aggregateDaily_empty_returnsZero() {
    when(taskMapper.countCompletedByUserBetween(anyLong(), any(), any())).thenReturn(0L);
    when(taskMapper.countByUserBetween(anyLong(), any(), any())).thenReturn(0L);
    when(taskMapper.statusDistribution(anyLong(), any(), any())).thenReturn(Map.of());
    when(taskMapper.priorityDistribution(anyLong(), any(), any())).thenReturn(Map.of());

    TaskMetrics m = provider.aggregateDaily(1L, LocalDate.of(2026, 7, 15));
    assertThat(m.completedCount()).isZero();
    assertThat(m.totalCount()).isZero();
    assertThat(m.completionRate()).isZero();
  }

  @Test void aggregateDaily_someCompleted_computesRate() {
    when(taskMapper.countCompletedByUserBetween(anyLong(), any(), any())).thenReturn(3L);
    when(taskMapper.countByUserBetween(anyLong(), any(), any())).thenReturn(5L);
    when(taskMapper.statusDistribution(anyLong(), any(), any())).thenReturn(Map.of("DONE", 3L, "TODO", 2L));
    when(taskMapper.priorityDistribution(anyLong(), any(), any())).thenReturn(Map.of("HIGH", 1L, "MEDIUM", 4L));

    TaskMetrics m = provider.aggregateDaily(1L, LocalDate.of(2026, 7, 15));
    assertThat(m.completionRate()).isEqualTo(0.6);
    assertThat(m.statusDistribution()).containsEntry("DONE", 3L);
  }
}
