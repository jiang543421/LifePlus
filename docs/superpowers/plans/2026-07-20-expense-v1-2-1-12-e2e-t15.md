```typescript
// frontend/e2e/expenses.spec.ts
import { test, expect } from '@playwright/test';

const loginStub = async (page: import('@playwright/test').Page) => {
  await page.route('**/api/v1/auth/login', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { accessToken: 'tok', refreshToken: 'ref' } }) }));
  await page.route('**/api/v1/users/me', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { id: 1, email: 'a@b.com', nickname: 'A' } }) }));
};

test.beforeEach(async ({ page }) => { await loginStub(page); });

test('list → summary renders', async ({ page }) => {
  await page.route('**/api/v1/expenses?**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { items: [], total: 0, page: 1, size: 20 } }) }));
  await page.route('**/api/v1/expenses/summary**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { totalAmount: '50.00', categoryBreakdown: { MEAL: '50.00' },
                monthOverMonthDelta: null, yearOverYearDelta: null } }) }));
  await page.route('**/api/v1/expenses/categories', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok', data: [
        { code: 'MEAL', label: '餐饮' }, { code: 'SHOPPING', label: '购物' },
        { code: 'TRANSPORT', label: '交通' }, { code: 'SUBSCRIPTION', label: '订阅' },
        { code: 'OTHER', label: '其他' } ]}) }));

  await page.goto('/expenses');
  await expect(page.locator('.summary-card .total')).toContainText('¥ 50.00');
});

test('delete confirms then removes', async ({ page }) => {
  await page.route('**/api/v1/expenses?**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { items: [{ id: 5, amount: '20.00', category: 'MEAL', note: null,
          occurredAt: '2026-07-15T12:00:00', createdAt: '', updatedAt: '' }],
          total: 1, page: 1, size: 20 } }) }));
  await page.route('**/api/v1/expenses/summary**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { totalAmount: '20.00', categoryBreakdown: { MEAL: '20.00' },
                monthOverMonthDelta: null, yearOverYearDelta: null } }) }));
  await page.route('**/api/v1/expenses/categories', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok', data: [] }) }));
  await page.route('**/api/v1/expenses/5', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok', data: null }) }));

  await page.goto('/expenses');
  page.once('dialog', d => d.accept());
  await page.locator('button:has-text("删除")').first().click();
  await expect(page.locator('.el-message--success')).toBeVisible();
});

test('unauthenticated → redirect /login', async ({ page }) => {
  await page.route('**/api/v1/users/me', route =>
    route.fulfill({ status: 401, body: '{"code":1002,"message":"未登录"}' }));
  await page.goto('/expenses');
  await expect(page).toHaveURL(/\/login/);
});
```

## 9. Risks & Open Questions
