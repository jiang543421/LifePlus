```typescript
import type { AiInsight } from '@/types/ai';
import { http } from './http';

export const aiApi = {
  async getToday(): Promise<AiInsight> {
    const r = await http.get<{ success: boolean; data: AiInsight }>('/api/v1/ai/insight/today');
    return r.data;
  },

  async getAnalysis(): Promise<AiInsight> {
    const r = await http.get<{ success: boolean; data: AiInsight }>('/api/v1/ai/insight/analysis');
    return r.data;
  },

  async refresh(): Promise<AiInsight> {
    const r = await http.post<{ success: boolean; data: AiInsight }>('/api/v1/ai/insight/refresh');
    return r.data;
  },
};
```

- [ ] **Step 2: stores/aiInsight.ts 加 refresh + getAnalysis**

```typescript
import { defineStore } from 'pinia';
import { ref } from 'vue';
import type { AiInsight } from '@/types/ai';
import { aiApi } from '@/api/ai';

export const useAiInsightStore = defineStore('aiInsight', () => {
  const insight = ref<AiInsight | null>(null);
  const loading = ref(false);
  const error = ref<Error | null>(null);

  async function loadToday() {
    loading.value = true;
    error.value = null;
    try {
      insight.value = await aiApi.getToday();
    } catch (e) {
      error.value = e as Error;
      insight.value = null;
    } finally {
      loading.value = false;
    }
  }

  async function loadAnalysis() {
    // 共享缓存：若已有数据且新鲜（< 6h），直接复用；否则 fetch
    if (insight.value && insight.value.freshnessSeconds < 6 * 3600) {
      return insight.value;
    }
    loading.value = true;
    try {
      insight.value = await aiApi.getAnalysis();
      return insight.value;
    } finally {
      loading.value = false;
    }
  }

  async function refresh() {
    loading.value = true;
    try {
      insight.value = await aiApi.refresh();
    } finally {
      loading.value = false;
    }
  }

  return { insight, loading, error, loadToday, loadAnalysis, refresh };
});
```

- [ ] **Step 3: Write store test**

`frontend/src/__tests__/stores/aiInsight.spec.ts`:

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { setActivePinia, createPinia } from 'pinia';
import { useAiInsightStore } from '@/stores/aiInsight';
import { aiApi } from '@/api/ai';

vi.mock('@/api/ai');

describe('aiInsight store', () => {
  beforeEach(() => setActivePinia(createPinia()));

  it('loadToday fetches and stores insight', async () => {
    const sample = { headline: 'h', chips: [], generatedAt: '2026-07-22T17:30:00Z', freshnessSeconds: 0, source: 'llm' } as any;
    vi.mocked(aiApi.getToday).mockResolvedValue(sample);
    const s = useAiInsightStore();
    await s.loadToday();
    expect(s.insight?.source).toBe('llm');
  });

  it('loadAnalysis reuses fresh insight without fetch', async () => {
    const s = useAiInsightStore();
    s.insight = { headline: 'h', chips: [], generatedAt: '2026-07-22T17:30:00Z', freshnessSeconds: 60 } as any;
    vi.mocked(aiApi.getAnalysis).mockClear();
    const r = await s.loadAnalysis();
    expect(r.headline).toBe('h');
    expect(aiApi.getAnalysis).not.toHaveBeenCalled();
  });

  it('refresh forces refetch and updates insight', async () => {
    const fresh = { headline: 'new', chips: [], generatedAt: '2026-07-22T18:00:00Z', freshnessSeconds: 0, source: 'template' } as any;
    vi.mocked(aiApi.refresh).mockResolvedValue(fresh);
    const s = useAiInsightStore();
    await s.refresh();
    expect(s.insight?.headline).toBe('new');
    expect(s.insight?.source).toBe('template');
  });
});
```
