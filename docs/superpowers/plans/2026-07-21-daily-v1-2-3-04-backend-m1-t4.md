```java
package com.lifepulse.daily.provider;

import com.lifepulse.daily.DailyConstants;
import com.lifepulse.daily.ExpenseMetrics;
import com.lifepulse.daily.ExpenseMetrics.CategoryTop;
import com.lifepulse.expense.repository.ExpenseMapper;
import org.springframework.stereotype.Component;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Comparator;
import java.util.List;
import java.util.Map;

@Component
public class ExpenseMetricProvider implements MetricProvider<ExpenseMetrics> {

  private final ExpenseMapper expenseMapper;

  public ExpenseMetricProvider(ExpenseMapper expenseMapper) {
    this.expenseMapper = expenseMapper;
  }

  @Override
  public ExpenseMetrics aggregateDaily(long userId, LocalDate date) {
    LocalDateTime from = date.atStartOfDay(DailyConstants.ZONE).toLocalDateTime();
    LocalDateTime to = date.plusDays(1).atStartOfDay(DailyConstants.ZONE).toLocalDateTime().minusNanos(1);
    BigDecimal total = expenseMapper.summaryTotal(userId, from, to);
    Map<String, BigDecimal> breakdown = expenseMapper.summaryByCategory(userId, from, to);
    long count = expenseMapper.countByUser(userId, null, from, to);
    List<CategoryTop> top = breakdown.entrySet().stream()
        .map(e -> new CategoryTop(e.getKey(), e.getValue()))
        .sorted(Comparator.comparing(CategoryTop::amount).reversed())
        .limit(3)
        .toList();
    return new ExpenseMetrics(
        total == null ? BigDecimal.ZERO : total,
        count,
        breakdown == null ? Map.of() : breakdown,
        top);
  }
}
```

@Test void aggregateDaily_empty_returnsZeros() {
  when(expenseMapper.summaryTotal(anyLong(), any(), any())).thenReturn(BigDecimal.ZERO);
  when(expenseMapper.summaryByCategory(anyLong(), any(), any())).thenReturn(Map.of());
  when(expenseMapper.countByUser(anyLong(), isNull(), any(), any())).thenReturn(0L);

  var m = provider.aggregateDaily(1L, LocalDate.of(2026, 7, 15));
  assertThat(m.totalAmount()).isEqualByComparingTo("0");
  assertThat(m.count()).isZero();
  assertThat(m.categoryBreakdown()).isEmpty();
  assertThat(m.topCategories()).isEmpty();
}

@Test void aggregateDaily_topThreeSorted() {
  when(expenseMapper.summaryTotal(anyLong(), any(), any())).thenReturn(new BigDecimal("100.00"));
  when(expenseMapper.summaryByCategory(anyLong(), any(), any())).thenReturn(Map.of(
      "MEAL", new BigDecimal("50"),
      "TRANSPORT", new BigDecimal("30"),
      "SHOPPING", new BigDecimal("20")));
  when(expenseMapper.countByUser(anyLong(), isNull(), any(), any())).thenReturn(10L);

  var m = provider.aggregateDaily(1L, LocalDate.of(2026, 7, 15));
  assertThat(m.topCategories()).hasSize(3);
  assertThat(m.topCategories().get(0).code()).isEqualTo("MEAL");
}
