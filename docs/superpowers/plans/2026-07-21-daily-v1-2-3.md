```java
package com.lifepulse.daily.provider;

import com.lifepulse.daily.DailyConstants;
import com.lifepulse.daily.TaskMetrics;
import com.lifepulse.task.repository.TaskMapper;
import org.springframework.stereotype.Component;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Map;

@Component
public class TaskMetricProvider implements MetricProvider<TaskMetrics> {

  private final TaskMapper taskMapper;

  public TaskMetricProvider(TaskMapper taskMapper) {
    this.taskMapper = taskMapper;
  }

  @Override
  public TaskMetrics aggregateDaily(long userId, LocalDate date) {
    LocalDateTime from = date.atStartOfDay(DailyConstants.ZONE).toLocalDateTime();
    LocalDateTime to = date.plusDays(1).atStartOfDay(DailyConstants.ZONE).toLocalDateTime().minusNanos(1);
    // 任务完成数 = 截止日在当天 OR 完成日在当天 且 status=DONE
    long completed = taskMapper.countCompletedByUserBetween(userId, from, to);
    long total = taskMapper.countByUserBetween(userId, from, to);
    Map<String, Long> status = taskMapper.statusDistribution(userId, from, to);
    Map<String, Long> priority = taskMapper.priorityDistribution(userId, from, to);
    double rate = total == 0 ? 0.0 : (double) completed / total;
    return new TaskMetrics(completed, total, rate, status, priority);
  }
}
```

> **TaskMapper 新增方法**(若不存在):`countCompletedByUserBetween` / `countByUserBetween` / `statusDistribution` / `priorityDistribution`。
> 若旧版 mapper 用 `dueDate` 字段,沿用既有查询,不强制新建,优先复用。

```java
package com.lifepulse.daily.provider;

import com.lifepulse.daily.DailyConstants;
import com.lifepulse.daily.ExpenseMetrics;
import com.lifepulse.daily.ExpenseMetrics.CategoryTop;
import com.lifepulse.expense.repository.ExpenseMapper;
import org.springframework.stereotype.Component;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Comparator;
import java.util.List;
import java.util.Map;

@Component
public class ExpenseMetricProvider implements MetricProvider<ExpenseMetrics> {

  private final ExpenseMapper expenseMapper;

  public ExpenseMetricProvider(ExpenseMapper expenseMapper) {
    this.expenseMapper = expenseMapper;
  }

  @Override
  public ExpenseMetrics aggregateDaily(long userId, LocalDate date) {
    LocalDateTime from = date.atStartOfDay(DailyConstants.ZONE).toLocalDateTime();
    LocalDateTime to = date.plusDays(1).atStartOfDay(DailyConstants.ZONE).toLocalDateTime().minusNanos(1);
    BigDecimal total = expenseMapper.summaryTotal(userId, from, to);
    Map<String, BigDecimal> breakdown = expenseMapper.summaryByCategory(userId, from, to);
    long count = expenseMapper.countByUser(userId, null, from, to);
    List<CategoryTop> top = breakdown.entrySet().stream()
        .map(e -> new CategoryTop(e.getKey(), e.getValue()))
        .sorted(Comparator.comparing(CategoryTop::amount).reversed())
        .limit(3)
        .toList();
    return new ExpenseMetrics(
        total == null ? BigDecimal.ZERO : total,
        count,
        breakdown == null ? Map.of() : breakdown,
        top);
  }
}
```

```java
package com.lifepulse.daily.service;

import com.lifepulse.common.exception.BusinessException;
import com.lifepulse.common.exception.ErrorCode;
import com.lifepulse.daily.*;
import com.lifepulse.daily.WeeklyComparison.WeeklyTriplet;
import com.lifepulse.daily.provider.*;
import com.lifepulse.security.UserContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;

import java.time.*;
import java.time.format.DateTimeParseException;
import java.time.temporal.WeekFields;
import java.util.Locale;

@Service
public class DailyReportService {

  private static final Logger log = LoggerFactory.getLogger(DailyReportService.class);

  private final TaskMetricProvider taskProvider;
  private final PlanMetricProvider planProvider;
  private final ExpenseMetricProvider expenseProvider;
  private final DietMetricProvider dietProvider;

  public DailyReportService(TaskMetricProvider taskProvider,
                             PlanMetricProvider planProvider,
                             ExpenseMetricProvider expenseProvider,
                             DietMetricProvider dietProvider) {
    this.taskProvider = taskProvider;
    this.planProvider = planProvider;
    this.expenseProvider = expenseProvider;
    this.dietProvider = dietProvider;
  }

  public DailyReportPayload aggregateDaily(LocalDate date) {
    Long userId = requireUserId();
    validateDate(date);
    log.debug("daily aggregate uid={} date={}", userId, date);
    return new DailyReportPayload(
        date,
        taskProvider.aggregateDaily(userId, date),
        planProvider.aggregateDaily(userId, date),
        expenseProvider.aggregateDaily(userId, date),
        dietProvider.aggregateDaily(userId, date));
  }

  public WeeklyReportPayload aggregateWeekly(String isoWeek) {
    Long userId = requireUserId();
    WeekFields wf = WeekFields.of(DailyConstants.WEEK_START, 4);
    LocalDate weekStart;
    try {
      int year = Integer.parseInt(isoWeek.substring(0, 4));
      int week = Integer.parseInt(isoWeek.substring(6));
      weekStart = LocalDate.now()
          .with(wf.weekBasedYear(year))
          .with(wf.weekOfWeekBasedYear(week))
          .with(wf.dayOfWeek() == 1 ? java.time.temporal.ChronoField.DAY_OF_WEEK ? null : null);
      // 简化:直接用 Year/WeekOfWeekBasedYear
      weekStart = LocalDate.of(year, 1, 4)
          .with(wf.weekOfWeekBasedYear(week))
          .with(wf.dayOfWeek(), 1);
    } catch (DateTimeParseException | NumberFormatException ex) {
      throw new BusinessException(ErrorCode.VALIDATION, "周格式非法: " + isoWeek);
    }
    LocalDate weekEnd = weekStart.plusDays(6);
    WeeklyComparison comp = computeComparison(userId, weekStart, weekEnd, wf);
    return new WeeklyReportPayload(isoWeek, weekStart, weekEnd, comp);
  }

  private WeeklyComparison computeComparison(Long userId, LocalDate weekStart, LocalDate weekEnd, WeekFields wf) {
    double taskCur = sumCompletion(userId, weekStart, weekStart.plusDays(6));
    double taskPrev = sumCompletion(userId, weekStart.minusDays(7), weekStart.minusDays(1));
    double planCur = sumCount(userId, weekStart, weekStart.plusDays(6), planProvider);
    double planPrev = sumCount(userId, weekStart.minusDays(7), weekStart.minusDays(1), planProvider);
    double expCur = sumExpense(userId, weekStart, weekStart.plusDays(6));
    double expPrev = sumExpense(userId, weekStart.minusDays(7), weekStart.minusDays(1));
    return new WeeklyComparison(
        triplet(taskCur, taskPrev),
        triplet(planCur, planPrev),
        triplet(expCur, expPrev));
  }

  private WeeklyTriplet triplet(double cur, double prev) {
    Double delta = prev == 0.0 ? null : cur - prev;
    return new WeeklyTriplet(cur, prev, delta);
  }

  // sum* helpers 略,模式相同:遍历 LocalDate,调用 provider,聚合
  // ...

  private Long requireUserId() {
    Long userId = UserContext.current();
    if (userId == null) throw new BusinessException(ErrorCode.BAD_CREDENTIALS, "未登录");
    return userId;
  }

  private void validateDate(LocalDate date) {
    LocalDate today = LocalDate.now(DailyConstants.ZONE);
    if (date.isAfter(today)) {
      throw new BusinessException(ErrorCode.VALIDATION, "不支持未来日期");
    }
    if (date.isBefore(today.minusDays(DailyConstants.MAX_HISTORY_DAYS))) {
      throw new BusinessException(ErrorCode.VALIDATION, "历史日期超出 " + DailyConstants.MAX_HISTORY_DAYS + " 天");
    }
  }
}
```

```java
public class DailyReportController {

  private static final Pattern ISO_WEEK = Pattern.compile("^\\d{4}-W\\d{2}$");

  private final DailyReportService service;

  public DailyReportController(DailyReportService service) {
    this.service = service;
  }

  @GetMapping
  public MyResponse<DailyReportPayload> daily(
      @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate date) {
    LocalDate target = date == null ? LocalDate.now() : date;
    return MyResponse.ok(service.aggregateDaily(target));
  }

  @GetMapping("/week")
  public MyResponse<WeeklyReportPayload> weekly(
      @RequestParam(required = false) String week) {
    String target = week == null ? currentIsoWeek() : week;
    if (!ISO_WEEK.matcher(target).matches()) {
      throw new BusinessException(ErrorCode.VALIDATION, "周格式非法,应为 YYYY-Www");
    }
    return MyResponse.ok(service.aggregateWeekly(target));
  }

  private String currentIsoWeek() {
    LocalDate today = LocalDate.now();
    int year = today.get(java.time.temporal.ChronoField.YEAR_OF_WEEK_BASED_YEAR);
    int w = today.get(java.time.temporal.ChronoField.WEEK_OF_WEEK_BASED_YEAR);
    return String.format("%d-W%02d", year, w);
  }
}
```

```typescript
// frontend/src/stores/daily.ts
import { defineStore } from 'pinia';
import dayjs from 'dayjs';
import { dailyApi } from '@/api/daily';
import type { DailyReportPayload, WeeklyReportPayload } from '@/types/daily';

export type ViewMode = 'daily' | 'weekly';
export interface DailyError { code: number; message: string }

export const useDailyStore = defineStore('daily', () => {
  const daily = ref<DailyReportPayload | null>(null);
  const weekly = ref<WeeklyReportPayload | null>(null);
  const currentDate = ref(dayjs().format('YYYY-MM-DD'));
  const currentWeek = ref('');  // 后端默认
  const viewMode = ref<ViewMode>('daily');
  const loading = ref(false);
  const error = ref<DailyError | null>(null);

  async function fetchDaily(date?: string) {
    loading.value = true; error.value = null;
    try {
      const target = date ?? currentDate.value;
      daily.value = await dailyApi.getDailyReport(target);
      currentDate.value = target;
    } catch (e: any) {
      error.value = { code: e.response?.data?.code ?? 500,
                      message: e.response?.data?.message ?? '加载失败' };
      throw e;
    } finally { loading.value = false; }
  }

  async function fetchWeekly(isoWeek?: string) {
    loading.value = true; error.value = null;
    try {
      const target = isoWeek ?? currentWeek.value;
      weekly.value = await dailyApi.getWeeklyReport(target);
      currentWeek.value = target;
    } catch (e: any) {
      error.value = { code: e.response?.data?.code ?? 500,
                      message: e.response?.data?.message ?? '加载失败' };
      throw e;
    } finally { loading.value = false; }
  }

  function setViewMode(mode: ViewMode) { viewMode.value = mode; }

  function setCurrentDate(d: string) { currentDate.value = d; }

  function shiftDate(days: number) {
    currentDate.value = dayjs(currentDate.value).add(days, 'day').format('YYYY-MM-DD');
  }

  function shiftWeek(weeks: number) {
    currentDate.value = dayjs(currentDate.value).add(weeks * 7, 'day').format('YYYY-MM-DD');
  }

  return { daily, weekly, currentDate, currentWeek, viewMode, loading, error,
           fetchDaily, fetchWeekly, setViewMode, setCurrentDate, shiftDate, shiftWeek };
});
```

> **Vue auto-import**:若项目用 `unplugin-auto-import`,`ref` 不需手动 import;否则顶部加 `import { ref } from 'vue'`。

## 8. Risks & Open Questions

| 风险 | 等级 | 处置 |
|---|---|---|
| 周报聚合 SQL 在大表(>10w 行)慢 | 中 | M0 索引必须命中;集成测试种 1w 行做 P95 断言;EXPLAIN 检查 |

| 索引变更锁表 | 低 | MySQL 8 INPLACE DDL;小表秒级;RELEASES 文档记录回滚 SQL |
| 饮食 Provider 误改 enabled=true | 低 | `DietMetricProviderTest` 锁死 enabled=false;PR 审阅卡 + IllegalStateException 运行时拦 |
| ECharts 等前端图表包体 | 不适用 | 日报不使用 ECharts(沿用 Element Plus 内置 Progress / Tag / List),首页卡不引入图表 |
| 周报边界被改成「滚动 7 天」 | 低 | 设计 §2.2 锁定 ISO 8601;US-3 AC 锁死周一为首 |
| 大用户(10w+ 行 t_task)首次加载日报超时 | 低 | 索引 + 分页(若需要);P95 监控;后续可加 Redis 缓存(留 E11) |
| 单开发者 6d 节奏不可预期 | 中 | 拆小任务到 ≤ 2d;每个可暂停点 commit;T1/T6/T10 为天然 checkpoint |
| M4 前端组件工作量超预期 | 中 | M3 完成后即开始 M4;若 spec 锁的组件细节过多,优先交付核心 3 个(任务/日程/消费),饮食/对比条可后置 |
