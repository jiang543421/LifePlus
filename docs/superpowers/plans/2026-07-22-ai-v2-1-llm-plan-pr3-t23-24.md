- [ ] **Step 1: Write the failing test**

```ts
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { mount } from '@vue/test-utils';
import { createPinia, setActivePinia } from 'pinia';
import AiAnalysisView from '../AiAnalysisView.vue';
import { useAiInsightStore } from '@/stores/aiInsight';

describe('AiAnalysisView', () => {
  beforeEach(() => { setActivePinia(createPinia()); });

  it('renders headline and 4 sections when payload present', async () => {
    const store = useAiInsightStore();
    store.analysis = {
      headline: '今日节奏平稳',
      advice: '保持当前节奏',
      highlight: '完成 3 项任务',
      mood: 'POSITIVE',
      source: 'llm',
      chips: [{ key: 'task', label: '任务', value: '3/5' }],
      generatedAt: '2026-07-22T10:00:00Z',
      freshnessSeconds: 0,
      llmMeta: null,
    };
    const wrapper = mount(AiAnalysisView, { global: { stubs: ['router-link'] } });
    expect(wrapper.text()).toContain('今日节奏平稳');
    expect(wrapper.text()).toContain('建议');
    expect(wrapper.text()).toContain('亮点');
    expect(wrapper.text()).toContain('任务');
  });

  it('shows [AI 生成] tag when source=llm', () => {
    const store = useAiInsightStore();
    store.analysis = { headline: 'x', advice: 'y', highlight: 'z', mood: 'POSITIVE', source: 'llm', chips: [], generatedAt: '', freshnessSeconds: 0, llmMeta: null };
    const wrapper = mount(AiAnalysisView, { global: { stubs: ['router-link'] } });
    expect(wrapper.text()).toContain('AI 生成');
  });

  it('shows degraded hint when source=template', () => {
    const store = useAiInsightStore();
    store.analysis = { headline: 'x', chips: [], source: 'template', generatedAt: '', freshnessSeconds: 0 };
    const wrapper = mount(AiAnalysisView, { global: { stubs: ['router-link'] } });
    expect(wrapper.text()).toContain('模板');
  });

  it('shows empty state when analysis is null', () => {
    const store = useAiInsightStore();
    store.analysis = null;
    const wrapper = mount(AiAnalysisView, { global: { stubs: ['router-link'] } });
    expect(wrapper.text()).toContain('暂无');
  });
});
```

- [ ] **Step 3: Create component**

`frontend/src/views/AiAnalysisView.vue`:

```vue
<template>
  <div class="ai-analysis">
    <header class="page-header">
      <h1>AI 分析</h1>
      <el-button :loading="store.loading" @click="store.fetchAnalysis(true)">
        刷新
      </el-button>
    </header>

    <div v-if="store.loading" v-loading="true" class="loading-block" />

    <el-empty
      v-else-if="!store.analysis"
      description="暂无分析数据，试试点击刷新"
    />

    <template v-else>
      <section class="section headline">
        <div class="headline-row">
          <h2>{{ store.analysis.headline }}</h2>
          <el-tag v-if="store.analysis.source === 'llm'" type="success" size="small">
            AI 生成
          </el-tag>
          <el-tag v-else type="info" size="small">
            模板生成
          </el-tag>
        </div>
        <p v-if="store.analysis.generatedAt" class="meta">
          生成于 {{ formatTime(store.analysis.generatedAt) }}
        </p>
      </section>

      <section class="section" aria-label="建议">
        <h3>建议</h3>
        <p>{{ store.analysis.advice || '（暂无建议）' }}</p>
      </section>

      <section class="section" aria-label="亮点">
        <h3>亮点</h3>
        <p>{{ store.analysis.highlight || '（暂无亮点）' }}</p>
        <span v-if="store.analysis.mood" class="mood">心情：{{ moodLabel(store.analysis.mood) }}</span>
      </section>

      <section class="section chips" aria-label="数据指标">
        <h3>今日数据</h3>
        <div class="chip-row">
          <div v-for="chip in store.analysis.chips" :key="chip.key" class="chip">
            <span class="chip-label">{{ chip.label }}</span>
            <span class="chip-value">{{ chip.value }}</span>
          </div>
        </div>
      </section>

      <el-alert
        v-if="store.analysis.source === 'template'"
        type="info"
        :closable="false"
        show-icon
      >
        AI 服务暂不可用，已展示模板结果
      </el-alert>
    </template>
  </div>
</template>

<script setup lang="ts">
import { onMounted } from 'vue';
import { useAiInsightStore } from '@/stores/aiInsight';

const store = useAiInsightStore();

onMounted(() => {
  if (!store.analysis) store.fetchAnalysis(false);
});

function moodLabel(m: string): string {
  return ({ POSITIVE: '积极', NEUTRAL: '平稳', NEGATIVE: '低落' } as Record<string, string>)[m] ?? m;
}

function formatTime(iso: string): string {
  return new Date(iso).toLocaleString('zh-CN', { hour12: false });
}
</script>
```
