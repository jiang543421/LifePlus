```vue
<!-- frontend/src/views/DailyReportView.vue -->
<script setup lang="ts">
import { onMounted } from 'vue';
import { useDailyStore } from '@/stores/daily';
import DailyReportHeader from '@/components/daily/DailyReportHeader.vue';
import TaskMetricsCard from '@/components/daily/TaskMetricsCard.vue';
import PlanMetricsCard from '@/components/daily/PlanMetricsCard.vue';
import ExpenseMetricsCard from '@/components/daily/ExpenseMetricsCard.vue';
import DietMetricsCard from '@/components/daily/DietMetricsCard.vue';
import WeeklyComparisonBar from '@/components/daily/WeeklyComparisonBar.vue';

const store = useDailyStore();

onMounted(async () => {
  if (store.viewMode === 'daily') await store.fetchDaily();
  else await store.fetchWeekly();
});
</script>

<template>
  <div class="daily-view">
    <DailyReportHeader />
    <div v-if="store.error" class="error-state">
      <el-alert :title="store.error.message" type="error" show-icon>
        <el-button v-if="store.error.code === 1004" @click="store.setCurrentDate(new Date().toISOString().slice(0,10)); store.fetchDaily()">
          回到今天
        </el-button>
      </el-alert>
    </div>
    <div v-else-if="store.viewMode === 'daily' && store.daily" class="cards-row">
      <TaskMetricsCard :metrics="store.daily.task" />
      <PlanMetricsCard :metrics="store.daily.plan" />
      <ExpenseMetricsCard :metrics="store.daily.expense" />
      <DietMetricsCard :metrics="store.daily.diet" />
    </div>
    <div v-else-if="store.viewMode === 'weekly' && store.weekly" class="weekly-section">
      <WeeklyComparisonBar :comparison="store.weekly.comparison" />
    </div>
  </div>
</template>
```

```typescript
// DailyReportView.spec.ts
import { mount } from '@vue/test-utils';
import { setActivePinia, createPinia } from 'pinia';
import DailyReportView from '../DailyReportView.vue';
import { useDailyStore } from '@/stores/daily';

vi.mock('@/api/daily');

describe('DailyReportView', () => {
  beforeEach(() => { setActivePinia(createPinia()); });

  it('mount daily mode → fetchDaily called', async () => {
    const { dailyApi } = await import('@/api/daily');
    vi.mocked(dailyApi.getDailyReport).mockResolvedValue({
      date: '2026-07-15', task: { completedCount: 0, totalCount: 0, completionRate: 0, statusDistribution: {}, priorityDistribution: {} },
      plan: { eventCount: 0, totalMinutes: 0, categoryDistribution: {}, busiestHour: null },
      expense: { totalAmount: '0', count: 0, categoryBreakdown: {}, topCategories: [] },
      diet: { enabled: false, value: null, reason: '饮食模块未上线' },
    });
    mount(DailyReportView, {
      global: { stubs: { DailyReportHeader: true, TaskMetricsCard: true, PlanMetricsCard: true,
                          ExpenseMetricsCard: true, DietMetricsCard: true, WeeklyComparisonBar: true } },
    });
    await new Promise(r => setTimeout(r, 0));
    expect(dailyApi.getDailyReport).toHaveBeenCalled();
  });

  it('error 1004 → renders error alert with back-to-today', async () => {
    const { dailyApi } = await import('@/api/daily');
    vi.mocked(dailyApi.getDailyReport).mockRejectedValue(
      Object.assign(new Error('validation'), { response: { data: { code: 1004, message: '未来日期' } } }));
    const w = mount(DailyReportView, {
      global: { stubs: { DailyReportHeader: true, TaskMetricsCard: true, PlanMetricsCard: true,
                          ExpenseMetricsCard: true, DietMetricsCard: true, WeeklyComparisonBar: true, ElAlert: true } },
    });
    await new Promise(r => setTimeout(r, 0));
    expect(w.text()).toContain('未来日期');
  });

  it('weekly mode → renders WeeklyComparisonBar', async () => {
    const { dailyApi } = await import('@/api/daily');
    vi.mocked(dailyApi.getWeeklyReport).mockResolvedValue({
      isoWeek: '2026-W29', weekStart: '2026-07-13', weekEnd: '2026-07-19',
      comparison: {
        taskCompletion: { current: 12, previous: 10, delta: 2 },
        planEvents: { current: 8, previous: 5, delta: 3 },
        expenseAmount: { current: 500, previous: 0, delta: null },
      },
    });
    const store = useDailyStore();
    store.setViewMode('weekly');
    const w = mount(DailyReportView, {
      global: { stubs: { DailyReportHeader: true, TaskMetricsCard: true, PlanMetricsCard: true,
                          ExpenseMetricsCard: true, DietMetricsCard: true, WeeklyComparisonBar: true } },
    });
    await new Promise(r => setTimeout(r, 0));
    expect(w.findComponent({ name: 'WeeklyComparisonBar-stub' }).exists()).toBe(true);
  });

  it('diet card always disabled regardless of payload', async () => {
    const w = mount(DailyReportView, {
      global: { stubs: { DailyReportHeader: true, TaskMetricsCard: true, PlanMetricsCard: true,
                          ExpenseMetricsCard: true, DietMetricsCard: true, WeeklyComparisonBar: true } },
    });
    const store = useDailyStore();
    store.daily = { date: '2026-07-15', task: null as any, plan: null as any, expense: null as any,
                    diet: { enabled: false, value: null, reason: '饮食模块未上线' } };
    expect(w.findComponent({ name: 'DietMetricsCard-stub' }).exists()).toBe(true);
  });

  it('skeleton displayed during loading', () => {
    const store = useDailyStore();
    store.loading = true;
    const w = mount(DailyReportView, {
      global: { stubs: { DailyReportHeader: true, TaskMetricsCard: true, PlanMetricsCard: true,
                          ExpenseMetricsCard: true, DietMetricsCard: true, WeeklyComparisonBar: true } },
    });
    expect(store.loading).toBe(true);
  });

  it('1003 not possible because no /{id} path', () => {
    // 设计层断言:路由表无 /api/daily/{id} → 越权攻击面天然不存在
    const router = require('@/router').default;
    const hasIdPath = router.options.routes.some((r: any) => /\/daily\/.+/.test(r.path));
    expect(hasIdPath).toBe(false);
  });
});
```
