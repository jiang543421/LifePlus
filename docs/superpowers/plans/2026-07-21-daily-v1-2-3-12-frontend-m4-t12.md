```vue
<!-- frontend/src/components/daily/DailyReportHeader.vue -->
<script setup lang="ts">
import { computed } from 'vue';
import { useDailyStore } from '@/stores/daily';
import dayjs from 'dayjs';

const store = useDailyStore();

const isToday = computed(() => store.currentDate === dayjs().format('YYYY-MM-DD'));
const isMinBoundary = computed(() =>
  dayjs(store.currentDate).isBefore(dayjs().subtract(30, 'day'), 'day'));
const isMaxBoundary = computed(() =>
  dayjs(store.currentDate).isAfter(dayjs(), 'day'));
</script>

<template>
  <div class="daily-header">
    <el-radio-group v-model="store.viewMode" @change="store.viewMode === 'daily' ? store.fetchDaily() : store.fetchWeekly()">
      <el-radio-button value="daily">日报</el-radio-button>
      <el-radio-button value="weekly">周报</el-radio-button>
    </el-radio-group>
    <el-date-picker
      v-if="store.viewMode === 'daily'"
      v-model="store.currentDate"
      type="date"
      :disabled-date="(d: any) => dayjs(d).isAfter(dayjs(), 'day') || dayjs(d).isBefore(dayjs().subtract(30, 'day'), 'day')"
      @change="store.fetchDaily()"
    />
    <span v-else class="week-display">{{ store.weekly?.isoWeek ?? '' }}</span>
    <el-button :disabled="isMinBoundary" @click="store.shiftDate(-1); store.fetchDaily()">◀ 上一天</el-button>
    <el-button :disabled="isMaxBoundary" @click="store.shiftDate(1); store.fetchDaily()">下一天 ▶</el-button>
  </div>
</template>
```

```vue
<!-- frontend/src/components/daily/PlanMetricsCard.vue -->
<script setup lang="ts">
import { computed } from 'vue';
import type { PlanMetrics } from '@/types/daily';
const props = defineProps<{ metrics: PlanMetrics }>();
const formattedTime = computed(() => {
  const h = Math.floor(props.metrics.totalMinutes / 60);
  const m = props.metrics.totalMinutes % 60;
  return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}`;
});
</script>
<template>
  <div class="metrics-card plan-card">
    <div class="card-title">日程</div>
    <div class="big-number">{{ formattedTime }}</div>
    <div class="stats">{{ metrics.eventCount }} 个事件{{ metrics.busiestHour != null ? `,最忙 ${metrics.busiestHour}:00` : '' }}</div>
  </div>
</template>
```

- [ ] **Step 6]: 创建 `WeeklyComparisonBar`

```vue
<!-- frontend/src/components/daily/WeeklyComparisonBar.vue -->
<script setup lang="ts">
import { computed } from 'vue';
import type { WeeklyComparison } from '@/types/daily';
const props = defineProps<{ comparison: WeeklyComparison }>();

function fmtDelta(t: { current: number; previous: number; delta: number | null }): string {
  if (t.delta == null) return '— (上周为 0)';
  const sign = t.delta >= 0 ? '+' : '';
  return `${sign}${t.delta.toFixed(1)}`;
}
</script>
<template>
  <div class="comparison-bar">
    <div class="comparison-row">
      <span class="label">任务完成率</span>
      <span class="delta">{{ fmtDelta(comparison.taskCompletion) }}</span>
    </div>
    <div class="comparison-row">
      <span class="label">日程事件数</span>
      <span class="delta">{{ fmtDelta(comparison.planEvents) }}</span>
    </div>
    <div class="comparison-row">
      <span class="label">消费总额</span>
      <span class="delta">{{ fmtDelta(comparison.expenseAmount) }}</span>
    </div>
  </div>
</template>
```
