```typescript
// frontend/e2e/daily/daily-report.spec.ts
import { test, expect } from '@playwright/test';

const loginStub = async (page: import('@playwright/test').Page) => {
  await page.route('**/api/v1/auth/login', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { accessToken: 'tok', refreshToken: 'ref' } }) }));
  await page.route('**/api/v1/users/me', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { id: 1, email: 'a@b.com', nickname: 'A' } }) }));
};

const dailyMock = async (page: import('@playwright/test').Page, body: any) => {
  await page.route('**/api/daily**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok', data: body }) }));
};

test.beforeEach(async ({ page }) => { await loginStub(page); });

test('home → daily 跳转', async ({ page }) => {
  await dailyMock(page, {
    date: '2026-07-15',
    task: { completedCount: 3, totalCount: 5, completionRate: 0.6, statusDistribution: {}, priorityDistribution: {} },
    plan: { eventCount: 2, totalMinutes: 240, categoryDistribution: {}, busiestHour: 14 },
    expense: { totalAmount: '47.00', count: 2, categoryBreakdown: {}, topCategories: [] },
    diet: { enabled: false, value: null, reason: '饮食模块未上线' },
  });
  await page.goto('/');
  await page.click('text=今日日报');
  await expect(page).toHaveURL(/\/daily/);
});

test('正常日显示完成率 + 饮食 disabled', async ({ page }) => {
  await dailyMock(page, {
    date: '2026-07-15',
    task: { completedCount: 3, totalCount: 5, completionRate: 0.6, statusDistribution: {}, priorityDistribution: {} },
    plan: { eventCount: 2, totalMinutes: 240, categoryDistribution: {}, busiestHour: 14 },
    expense: { totalAmount: '47.00', count: 2, categoryBreakdown: {}, topCategories: [] },
    diet: { enabled: false, value: null, reason: '饮食模块未上线' },
  });
  await page.goto('/daily');
  await expect(page.locator('.task-card .big-number')).toContainText('60%');
  await expect(page.locator('.diet-card')).toHaveClass(/disabled/);
});

test('空数据日零值展示', async ({ page }) => {
  await dailyMock(page, {
    date: '2026-07-15',
    task: { completedCount: 0, totalCount: 0, completionRate: 0, statusDistribution: {}, priorityDistribution: {} },
    plan: { eventCount: 0, totalMinutes: 0, categoryDistribution: {}, busiestHour: null },
    expense: { totalAmount: '0.00', count: 0, categoryBreakdown: {}, topCategories: [] },
    diet: { enabled: false, value: null, reason: '饮食模块未上线' },
  });
  await page.goto('/daily');
  await expect(page.locator('.expense-card .big-number')).toContainText('¥ 0.00');
});

test('未来日 1004 → 回到今天按钮', async ({ page }) => {
  await page.route('**/api/daily**', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 1004, message: '未来日期', data: null }) }));
  await page.goto('/daily?date=2099-01-01');
  await expect(page.locator('.error-state')).toBeVisible();
  await expect(page.locator('button:has-text("回到今天")')).toBeVisible();
});

test('周报显示 3 组对比', async ({ page }) => {
  await dailyMock(page, {
    isoWeek: '2026-W29', weekStart: '2026-07-13', weekEnd: '2026-07-19',
    comparison: {
      taskCompletion: { current: 12, previous: 10, delta: 2 },
      planEvents: { current: 8, previous: 5, delta: 3 },
      expenseAmount: { current: 500, previous: 0, delta: null },
    },
  });
  await page.goto('/daily/week?week=2026-W29');
  await expect(page.locator('.comparison-row')).toHaveCount(3);
});

test('跨用户隔离 — user B 看不到 user A 数据', async ({ page }) => {
  // user B login
  await page.route('**/api/v1/users/me', route =>
    route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok',
        data: { id: 2, email: 'b@b.com', nickname: 'B' } }) }));
  await dailyMock(page, {
    date: '2026-07-15',
    task: { completedCount: 0, totalCount: 0, completionRate: 0, statusDistribution: {}, priorityDistribution: {} },
    plan: { eventCount: 0, totalMinutes: 0, categoryDistribution: {}, busiestHour: null },
    expense: { totalAmount: '0.00', count: 0, categoryBreakdown: {}, topCategories: [] },
    diet: { enabled: false, value: null, reason: '饮食模块未上线' },
  });
  await page.goto('/daily');
  await expect(page.locator('.task-card .big-number')).toContainText('0%');
});

test('上一天 / 下一天按钮工作', async ({ page }) => {
  let receivedDate = '';
  await page.route('**/api/daily**', route => {
    const url = new URL(route.request().url());
    receivedDate = url.searchParams.get('date') ?? '';
    return route.fulfill({ status: 200, contentType: 'application/json',
      body: JSON.stringify({ code: 0, message: 'ok', data: {
        date: receivedDate,
        task: { completedCount: 0, totalCount: 0, completionRate: 0, statusDistribution: {}, priorityDistribution: {} },
        plan: { eventCount: 0, totalMinutes: 0, categoryDistribution: {}, busiestHour: null },
        expense: { totalAmount: '0', count: 0, categoryBreakdown: {}, topCategories: [] },
        diet: { enabled: false, value: null, reason: '饮食模块未上线' },
      } }) });
  });
  await page.goto('/daily');
  await page.click('button:has-text("下一天")');
  expect(receivedDate).toBeTruthy();
});

test('未登录 → 跳 /login', async ({ page }) => {
  await page.route('**/api/v1/users/me', route =>
    route.fulfill({ status: 401, body: '{"code":1002,"message":"未登录"}' }));
  await page.goto('/daily');
  await expect(page).toHaveURL(/\/login/);
});
```
