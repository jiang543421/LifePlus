import { loginAs } from './helpers/auth';

test.describe('AI analysis independent page', () => {
  test.beforeEach(async ({ page }) => {
    await loginAs(page, 'alice@lp.dev', 'Passw0rd!');
  });

  test('home → drawer → independent page flow', async ({ page }) => {
    await page.goto('/');
    await page.getByTestId('ai-insight-card').click();
    await page.getByText('查看完整分析').click();
    await expect(page).toHaveURL(/\/ai-analysis$/);
    await expect(page.getByRole('heading', { name: 'AI 分析' })).toBeVisible();
  });

  test('independent page shows 4 sections + chips', async ({ page }) => {
    await page.goto('/ai-analysis');
    await expect(page.getByText(/建议|亮点|今日数据/)).toBeVisible();
  });

  test('shows degraded hint when LLM is down', async ({ page }) => {
    await page.route('**/api/ai/analysis', (route) =>
      route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          success: true,
          data: {
            headline: '今日节奏平稳',
            chips: [],
            source: 'template',
            generatedAt: new Date().toISOString(),
            freshnessSeconds: 0,
          },
        }),
      })
    );
    await page.goto('/ai-analysis');
    await expect(page.getByText('AI 服务暂不可用')).toBeVisible();
  });
});

- [ ] **Step 2: Write ai-llm.spec.ts（refresh 路径 + quota）**

```ts
import { test, expect } from '@playwright/test';
import { loginAs } from './helpers/auth';

test.describe('AI refresh + quota', () => {
  test.beforeEach(async ({ page }) => {
    await loginAs(page, 'alice@lp.dev', 'Passw0rd!');
  });

  test('refresh button triggers POST /api/ai/refresh and updates timestamp', async ({ page }) => {
    await page.goto('/');
    const before = await page.getByTestId('ai-meta-time').textContent();
    await page.getByTestId('ai-insight-refresh').click();
    await page.waitForResponse((r) => r.url().includes('/api/ai/refresh'));
    const after = await page.getByTestId('ai-meta-time').textContent();
    expect(after).not.toEqual(before);
  });

  test('quota exceeded returns 1510 with friendly hint', async ({ page }) => {
    await page.route('**/api/ai/refresh', (route) =>
      route.fulfill({
        status: 429,
        contentType: 'application/json',
        body: JSON.stringify({ success: false, error: { code: 1510, message: '今日 AI 额度已用完' } }),
      })
    );
    await page.goto('/');
    await page.getByTestId('ai-insight-refresh').click();
    await expect(page.getByText('今日 AI 额度已用完')).toBeVisible();
  });
});
```
