```java
package com.lifepulse.it;

import com.lifepulse.auth.AuthTestSupport;
import com.lifepulse.daily.*;
import com.lifepulse.expense.dto.CreateExpenseRequest;
import com.lifepulse.expense.service.ExpenseService;
import com.lifepulse.daily.service.DailyReportService;
import com.lifepulse.plan.dto.CreatePlanRequest;
import com.lifepulse.plan.service.PlanService;
import com.lifepulse.security.UserContext;
import com.lifepulse.task.dto.CreateTaskRequest;
import com.lifepulse.task.service.TaskService;
import org.junit.jupiter.api.*;
import org.springframework.beans.factory.annotation.Autowired;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.List;

import static org.assertj.core.api.Assertions.*;

class DailyReportIT extends AbstractIntegrationTest {

  @Autowired DailyReportService dailyService;
  @Autowired TaskService taskService;
  @Autowired PlanService planService;
  @Autowired ExpenseService expenseService;
  @Autowired AuthTestSupport auth;

  Long userA, userB;

  @BeforeEach
  void setUp() {
    auth.resetDatabase();
    userA = auth.register("a@example.com", "Password1", "A");
    userB = auth.register("b@example.com", "Password1", "B");
    UserContext.set(userA);
  }

  @AfterEach
  void tearDown() { UserContext.clear(); }

  @Test void aggregateDaily_empty_returnsZeroValues() {
    DailyReportPayload p = dailyService.aggregateDaily(LocalDate.of(2026, 7, 15));
    assertThat(p.task().totalCount()).isZero();
    assertThat(p.plan().eventCount()).isZero();
    assertThat(p.expense().totalAmount()).isEqualByComparingTo("0");
    assertThat(p.diet().enabled()).isFalse();
  }

  @Test void aggregateDaily_crossUser_isolated() {
    // userA 创建 1 笔 task
    taskService.create(new CreateTaskRequest("A的任务", "HIGH", null, null,
        LocalDateTime.of(2026, 7, 15, 12, 0)));
    // userB 切换后查询同一天,应为空
    UserContext.set(userB);
    DailyReportPayload p = dailyService.aggregateDaily(LocalDate.of(2026, 7, 15));
    assertThat(p.task().totalCount()).isZero();
  }

  @Test void aggregateDaily_softDeletedTask_excluded() {
    Long id = taskService.create(new CreateTaskRequest("软删任务", "HIGH", null, null,
        LocalDateTime.of(2026, 7, 15, 12, 0))).id();
    taskService.softDelete(id);
    DailyReportPayload p = dailyService.aggregateDaily(LocalDate.of(2026, 7, 15));
    assertThat(p.task().totalCount()).isZero();
  }

  @Test void aggregateWeekly_happy_returnsComparison() {
    // 本周一笔 + 上周一笔,验证 delta 计算
    taskService.create(new CreateTaskRequest("本周", "HIGH", null, null,
        LocalDateTime.of(2026, 7, 15, 12, 0)));
    UserContext.set(userA);
    WeeklyReportPayload p = dailyService.aggregateWeekly("2026-W29");
    assertThat(p.isoWeek()).isEqualTo("2026-W29");
    assertThat(p.weekStart()).isEqualTo(LocalDate.of(2026, 7, 13));
    assertThat(p.weekEnd()).isEqualTo(LocalDate.of(2026, 7, 19));
  }

  @Test void perfDaily_p95BudgetUnder200ms() {
    // 种 1w 行 task
    LocalDate today = LocalDate.now();
    for (int i = 0; i < DailyConstants.PERF_SEED_ROWS; i++) {
      taskService.create(new CreateTaskRequest("perf-" + i, "MEDIUM", null, null,
          LocalDateTime.of(today, LocalDateTime.now().toLocalTime())));
    }
    // 跑 100 次取 P95
    long[] elapsedNanos = new long[100];
    for (int i = 0; i < 100; i++) {
      long start = System.nanoTime();
      dailyService.aggregateDaily(today);
      elapsedNanos[i] = System.nanoTime() - start;
    }
    java.util.Arrays.sort(elapsedNanos);
    long p95Nanos = elapsedNanos[95];
    assertThat(p95Nanos)
        .as("Daily P95")
        .isLessThan(DailyConstants.DAILY_P95_BUDGET.toNanos());
  }
}
```

> **种子规模调整**:若 `PERF_SEED_ROWS=10_000` 让 CI 跑得太慢,在 Application-IT profile 把其降为 1_000;P95 阈值同步按比例放宽。
