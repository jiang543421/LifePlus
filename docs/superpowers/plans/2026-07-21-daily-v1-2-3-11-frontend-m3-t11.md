```vue
<el-col :xs="24" :sm="12" :md="8">
  <ModuleCard
    title="今日日报"
    :amount="`完成率 ${(dailySummary?.completionRate ?? 0 * 100).toFixed(0)}%`"
    :items="dailySummary?.topCategories ?? []"
    cta="/daily"
    cta-label="查看日报 →"
    @click="router.push('/daily')"
  />
</el-col>
```

```typescript
// <script setup> 新增
import { dailyApi } from '@/api/daily';
const dailySummary = ref<{ completionRate: number; topCategories: any[] } | null>(null);
onMounted(async () => {
  const today = dayjs().format('YYYY-MM-DD');
  const r = await dailyApi.getDailyReport(today);
  dailySummary.value = { completionRate: r.task.completionRate * 100, topCategories: [] };
});
```
