# AI 分析模块 v2.1 (LLM 增强) — 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 v2.0 智能卡 + 抽屉基础上叠加 LLM 洞察升级，并兑现 v2.0 留的 `/ai-analysis` 独立分析页承诺；通过 3 层降级（L1 LLM → L2 模板 → L3 1501）保证 v2.0 模板引擎仍为最终回退。

**Architecture:**
- 后端：v2.0 Service 注入 `LlmInsightGenerator`（手写 `RestClient` 调 OpenAI 兼容协议），Provider 聚合层与缓存层零改动。
- Redis 命名空间扩展：洞察缓存 TTL 30min→6h；新增 `lp:ai:quota:*` 配额键 + `lp:ai:circuit:*` 熔断键。
- 前端：智能卡加 `source` 标签；抽屉底部加"查看完整分析 →"路由；新增 `AiAnalysisView.vue`。
- 零 DB schema 变更、零 Maven 依赖变更、零 Spring Bean 冲突。

**Tech Stack:**
- 后端：Spring Boot 3.3.5 + Spring `RestClient`（Boot 内置）+ Spring Data Redis + JUnit 5 + Mockito + Testcontainers
- 前端：Vue 3.4 + TypeScript 5 + Element Plus 2.8 + Pinia 2 + Axios 1.7 + Vitest + Playwright
- LLM：DeepSeek V3 (`deepseek-chat`) 主选 + Ollama (`deepseek-r1:8b`) 备选（配置切换）

```java
package com.lifepulse.ai.llm;

import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotBlank;
import org.hibernate.validator.constraints.URL;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.stereotype.Component;
import org.springframework.validation.annotation.Validated;

/**
 * LLM provider configuration (spec §10.3 / CLAUDE.md §11.2).
 * Startup fail-fast on placeholder / short apiKey when enabled + deepseek.
 */
@Component
@ConfigurationProperties("lp.ai.llm")
@Validated
public record LlmProperties(
    boolean enabled,
    @NotBlank String provider,
    @URL String baseUrl,
    String apiKey,
    @NotBlank String model,
    @Min(1000) @Max(30000) int timeoutMs,
    @Min(100) @Max(4000) int maxPromptTokens,
    @Min(50) @Max(1000) int maxResponseTokens,
    @Min(1) @Max(1000) int dailyQuota,
    CircuitBreaker circuitBreaker
) {
    public LlmProperties {
        if (enabled && "deepseek".equals(provider)) {
            if (apiKey == null || apiKey.isBlank()) {
                throw new IllegalStateException(
                    "lp.ai.llm.api-key is required when provider=deepseek. " +
                    "Set LP_LLM_API_KEY env var or disable llm (lp.ai.llm.enabled=false).");
            }
            if (apiKey.startsWith("sk-replace-") || apiKey.length() < 20) {
                throw new IllegalStateException(
                    "lp.ai.llm.api-key looks like a placeholder. " +
                    "Replace LP_LLM_API_KEY with a real DeepSeek key.");
            }
        }
        if ("ollama".equals(provider) && circuitBreaker.enabled()) {
            // Ollama mode: circuit breaker disabled (local process semantics differ).
            // Build a copy with enabled=false; @ConfigurationProperties will not see this.
            // Logged in @PostConstruct instead.
        }
    }

    public record CircuitBreaker(
        boolean enabled,
        @Min(1) @Max(100) int failureThreshold,
        @Min(1) @Max(60) int windowMinutes,
        @Min(1) @Max(1440) int cooldownMinutes
    ) {}
}
```

```java
import org.springframework.web.client.ResourceAccessException;
import org.springframework.web.client.RestClient;
import org.springframework.web.client.RestClientException;

/**
 * DeepSeek V3 client (OpenAI-compatible protocol).
 * Conditional on lp.ai.llm.provider=deepseek.
 */
@Component
@ConditionalOnProperty(prefix = "lp.ai.llm", name = "provider", havingValue = "deepseek")
public class DeepSeekClient implements LlmClient {

    private static final Logger log = LoggerFactory.getLogger(DeepSeekClient.class);

    private final RestClient restClient;
    private final LlmProperties props;
    private final ObjectMapper objectMapper;

    public DeepSeekClient(RestClient.Builder builder, LlmProperties props, ObjectMapper objectMapper) {
        this.props = props;
        this.objectMapper = objectMapper;
        this.restClient = builder
            .baseUrl(props.baseUrl())
            .defaultHeader(HttpHeaders.AUTHORIZATION, "Bearer " + props.apiKey())
            .defaultHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
            .build();
    }

    @Override
    public LlmResponse generate(LlmRequest request) {
        long start = System.currentTimeMillis();
        String body = buildBody(request);
        try {
            String responseBody = restClient.post()
                .uri("/chat/completions")
                .body(body)
                .retrieve()
                .onStatus(HttpStatusCode::is4xxClientError, (req, res) -> {
                    int code = res.getStatusCode().value();
                    if (code == 429) {
                        throw new LlmUnavailableException("DeepSeek 429 rate limited");
                    }
                    throw new LlmResponseInvalidException("DeepSeek 4xx: " + code);
                })
                .onStatus(HttpStatusCode::is5xxServerError, (req, res) -> {
                    throw new LlmUnavailableException("DeepSeek 5xx: " + res.getStatusCode().value());
                })
                .body(String.class);

            return parse(responseBody, System.currentTimeMillis() - start);
        } catch (ResourceAccessException ex) {
            // timeout / connection failure
            throw new LlmUnavailableException("DeepSeek network failure: " + ex.getMessage(), ex);
        } catch (LlmUnavailableException | LlmResponseInvalidException ex) {
            throw ex;
        } catch (RestClientException ex) {
            throw new LlmUnavailableException("DeepSeek rest failure: " + ex.getMessage(), ex);
        }
    }

    private String buildBody(LlmRequest req) {
        return String.format("""
            {"model":"%s","max_tokens":%d,"messages":[
              {"role":"system","content":%s},
              {"role":"user","content":%s}
            ]}""",
            props.model(),
            req.maxResponseTokens(),
            jsonString(req.systemPrompt()),
            jsonString(req.userPrompt()));
    }

    private String jsonString(String s) {
        return "\"" + s.replace("\\", "\\\\").replace("\"", "\\\"") + "\"";
    }

    private LlmResponse parse(String body, long latencyMs) {
        if (body == null || body.isBlank()) {
            throw new LlmResponseInvalidException("Empty DeepSeek response body");
        }
        try {
            JsonNode root = objectMapper.readTree(body);
            String content = root.path("choices").path(0).path("message").path("content").asText(null);
            if (content == null) {
                throw new LlmResponseInvalidException("Missing choices[0].message.content");
            }
            int promptTokens = root.path("usage").path("prompt_tokens").asInt(0);
            int completionTokens = root.path("usage").path("completion_tokens").asInt(0);
            return new LlmResponse(content, promptTokens, completionTokens, latencyMs);
        } catch (com.fasterxml.jackson.core.JsonProcessingException ex) {
            throw new LlmResponseInvalidException("DeepSeek JSON parse failed: " + ex.getMessage(), ex);
        }
    }
}
```

```java
import org.slf4j.Logger;

public enum Mood {

    public static Mood fromString(String s) {
```

```java
import com.lifepulse.ai.AiConstants;
import com.lifepulse.ai.model.MetricValue;
import org.springframework.core.io.ClassPathResource;
import org.springframework.stereotype.Component;
import java.io.IOException;
import java.io.InputStreamReader;
import java.nio.charset.StandardCharsets;
import java.time.Duration;
import java.time.LocalDate;
import java.util.List;
import java.util.Properties;

@Component
public class LlmPromptBuilder {

    private final Properties props = loadProps();

    public LlmRequest build(long userId, List<MetricValue> metrics, LocalDate today) {
        String system = props.getProperty("system.role", "").trim();
        String user = renderUserPrompt(metrics, today);
        int maxTokens = parseInt(props.getProperty("max.response.tokens", "300"));
        Duration timeout = Duration.ofMillis(parseInt(props.getProperty("timeout.ms", "5000")));
        return new LlmRequest(system, user, maxTokens, timeout);
    }

    private String renderUserPrompt(List<MetricValue> metrics, LocalDate today) {
        if (metrics.isEmpty()) {
            return props.getProperty("user.empty", "暂无数据");
        }
        StringBuilder sb = new StringBuilder();
        for (MetricValue mv : metrics) {
            String key = mapKey(mv);
            sb.append(props.getProperty("user.chip." + key, "")).append("\n");
        }
        String template = props.getProperty("user.template", "{0}");
        return template.replace("{0}", sb.toString())
            .replace("{date}", today.toString());
    }

    private String mapKey(MetricValue mv) {
        // metric value 类型不可用，简化：用 value 非空 + trend 推测
        // 实际接入时由 Service 按 provider key 排序传入
        if (mv.value() == null) return "empty";
        return "taskCompletion";
    }

    private static int parseInt(String s) {
        try { return Integer.parseInt(s.trim()); } catch (NumberFormatException e) { return 0; }
    }

    private static Properties loadProps() {
        Properties p = new Properties();
        try (var in = new InputStreamReader(
                new ClassPathResource("llm-prompt.properties").getInputStream(),
                StandardCharsets.UTF_8)) {
            p.load(in);
        } catch (IOException ex) {
            throw new IllegalStateException("Failed to load llm-prompt.properties", ex);
        }
        return p;
    }
}
```

```java
package com.lifepulse.ai.llm;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.lifepulse.ai.llm.exception.LlmResponseInvalidException;
import org.springframework.stereotype.Component;
import java.util.Set;

@Component
public class LlmJsonParser {

    private static final int MIN_HEADLINE = 20;
    private static final int MAX_HEADLINE = 200;
    private static final int MIN_ADVICE = 10;
    private static final int MAX_ADVICE = 200;
    private static final int MIN_HIGHLIGHT = 10;
    private static final int MAX_HIGHLIGHT = 200;

    private static final Set<String> SENSITIVE_TOKENS = Set.of(
        "fuck", "shit", "傻逼", "操你妈"
    );

    private final ObjectMapper objectMapper;

    public LlmJsonParser(ObjectMapper objectMapper) {
        this.objectMapper = objectMapper;
    }

    public LlmInsightPayload parse(LlmResponse response) {
        String body = response.content();
        if (body == null || body.isBlank()) {
            throw new LlmResponseInvalidException("Empty LLM response body");
        }
        JsonNode root;
        try {
            root = objectMapper.readTree(body);
        } catch (Exception ex) {
            throw new LlmResponseInvalidException("LLM JSON parse failed: " + ex.getMessage(), ex);
        }
        String headline = requireString(root, "headline");
        String advice = requireString(root, "advice");
        String highlight = requireString(root, "highlight");
        validateLength("headline", headline, MIN_HEADLINE, MAX_HEADLINE);
        validateLength("advice", advice, MIN_ADVICE, MAX_ADVICE);
        validateLength("highlight", highlight, MIN_HIGHLIGHT, MAX_HIGHLIGHT);
        checkSensitive("advice", advice);
        checkSensitive("highlight", highlight);
        Mood mood = Mood.fromString(root.path("mood").asText(null));
        return new LlmInsightPayload(
            headline.trim(), advice.trim(), highlight.trim(), mood,
            response.promptTokens(), response.responseTokens(), response.latencyMs()
        );
    }

    private static String requireString(JsonNode root, String field) {
        String v = root.path(field).asText(null);
        if (v == null || v.isBlank()) {
            throw new LlmResponseInvalidException("Missing or empty field: " + field);
        }
        return v;
    }

    private static void validateLength(String field, String value, int min, int max) {
        int len = value.length();
        if (len < min || len > max) {
            throw new LlmResponseInvalidException(
                field + " length out of range [" + min + "," + max + "]: " + len);
        }
    }

    private static void checkSensitive(String field, String value) {
        String lower = value.toLowerCase();
        for (String token : SENSITIVE_TOKENS) {
            if (lower.contains(token)) {
                throw new LlmResponseInvalidException(
                    field + " contains sensitive token: " + token);
            }
        }
    }
}
```

```java
    private static final Logger log = LoggerFactory.getLogger(LlmQuotaGuard.class);

        this.dailyQuota = dailyQuota;
    }

    public void checkAndIncrement(long userId) {

            }
        } catch (RedisConnectionFailureException ex) {
            log.warn("LlmQuotaGuard: redis unavailable, fail-open for userId={}", userId);
        }
```

```java
package com.lifepulse.ai.llm;

import com.lifepulse.ai.llm.exception.LlmCircuitOpenException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.data.redis.RedisConnectionFailureException;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Component;
import java.time.Duration;
import java.time.Instant;
import java.util.concurrent.TimeUnit;

/**
 * LLM circuit breaker (spec §6.4 / CLAUDE.md §11.5).
 * - 5min sliding window of failure timestamps (ZSET)
 * - threshold failures -> OPEN for cooldownMinutes
 * - Ollama mode: enabled=false (local process semantics differ)
 * - Redis fail-closed: when Redis unavailable, do not trip
 */
@Component
public class LlmCircuitBreaker {

    private static final Logger log = LoggerFactory.getLogger(LlmCircuitBreaker.class);

    private static final String STATE_KEY = "lp:ai:circuit:state";
    private static final String OPENED_AT_KEY = "lp:ai:circuit:state:openedAt";
    private static final String FAILURES_KEY = "lp:ai:circuit:failures";

    private final StringRedisTemplate redis;
    private final boolean enabled;
    private final int failureThreshold;
    private final long windowMs;
    private final long cooldownMs;

    public LlmCircuitBreaker(StringRedisTemplate redis, LlmProperties props) {
        this.redis = redis;
        this.enabled = props.circuitBreaker().enabled();
        this.failureThreshold = props.circuitBreaker().failureThreshold();
        this.windowMs = Duration.ofMinutes(props.circuitBreaker().windowMinutes()).toMillis();
        this.cooldownMs = Duration.ofMinutes(props.circuitBreaker().cooldownMinutes()).toMillis();
    }

    public void tryAcquire(long userId) {
        if (!enabled) return;

        try {
            String state = redis.opsForValue().get(STATE_KEY);
            if ("OPEN".equals(state)) {
                long openedAt = parseLong(redis.opsForValue().get(OPENED_AT_KEY));
                long now = Instant.now().toEpochMilli();
                if (now - openedAt < cooldownMs) {
                    throw new LlmCircuitOpenException();
                }
                // cooldown expired -> HALF_OPEN
                redis.opsForValue().set(STATE_KEY, "HALF_OPEN");
            }

            long now = Instant.now().toEpochMilli();
            redis.opsForZSet().removeRangeByScore(FAILURES_KEY, 0, now - windowMs);
            Long failures = redis.opsForZSet().zCard(FAILURES_KEY);
            if (failures != null && failures >= failureThreshold) {
                redis.opsForValue().set(STATE_KEY, "OPEN");
                redis.opsForValue().set(OPENED_AT_KEY, String.valueOf(now));
                throw new LlmCircuitOpenException();
            }
        } catch (RedisConnectionFailureException ex) {
            log.warn("LlmCircuitBreaker: redis unavailable, fail-closed for userId={}", userId);
        }
    }

    public void recordSuccess() {
        if (!enabled) return;
        try {
            redis.opsForValue().set(STATE_KEY, "CLOSED");
            redis.delete(OPENED_AT_KEY);
            redis.delete(FAILURES_KEY);
        } catch (RedisConnectionFailureException ex) {
            log.warn("LlmCircuitBreaker.recordSuccess: redis unavailable, ignored", ex);
        }
    }

    public void recordFailure() {
        if (!enabled) return;
        try {
            long now = Instant.now().toEpochMilli();
            redis.opsForZSet().add(FAILURES_KEY, String.valueOf(now), now);
            redis.expire(FAILURES_KEY, Duration.ofMinutes(5));
        } catch (RedisConnectionFailureException ex) {
            log.warn("LlmCircuitBreaker.recordFailure: redis unavailable, ignored", ex);
        }
    }

    private static long parseLong(String s) {
        try { return s == null ? 0L : Long.parseLong(s.trim()); }
        catch (NumberFormatException e) { return 0L; }
    }
}
```

```java
package com.lifepulse.ai.llm;

import com.lifepulse.ai.model.MetricValue;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;
import java.time.LocalDate;
import java.util.List;

/**
 * LLM business orchestrator (spec §5 / CLAUDE.md §11.3).
 * Pure orchestration: quota -> circuit -> prompt -> client -> parser.
 * All 4 Llm*Exception types thrown as-is; Service decides whether to fall back.
 */
@Service
public class LlmInsightGenerator {

    private static final Logger log = LoggerFactory.getLogger(LlmInsightGenerator.class);

    private final LlmClient client;
    private final LlmJsonParser parser;
    private final LlmQuotaGuard quota;
    private final LlmCircuitBreaker breaker;
    private final LlmPromptBuilder promptBuilder;
    private final boolean enabled;

    public LlmInsightGenerator(LlmClient client, LlmJsonParser parser,
                                LlmQuotaGuard quota, LlmCircuitBreaker breaker,
                                LlmPromptBuilder promptBuilder, LlmProperties props) {
        this.client = client;
        this.parser = parser;
        this.quota = quota;
        this.breaker = breaker;
        this.promptBuilder = promptBuilder;
        this.enabled = props.enabled();
    }

    /**
     * @throws LlmQuotaExceededException    1510
     * @throws LlmCircuitOpenException     1511
     * @throws LlmResponseInvalidException 1512
     * @throws LlmUnavailableException     1513
     */
    public LlmInsightPayload generate(long userId, List<MetricValue> metrics, LocalDate today) {
        if (!enabled) {
            throw new IllegalStateException("LLM is disabled (lp.ai.llm.enabled=false)");
        }

        quota.checkAndIncrement(userId);
        breaker.tryAcquire(userId);

        LlmRequest request = promptBuilder.build(userId, metrics, today);
        LlmResponse response;
        try {
            response = client.generate(request);
            breaker.recordSuccess();
        } catch (RuntimeException ex) {
            breaker.recordFailure();
            throw ex;
        }
        return parser.parse(response);
    }
}
```

```java
        this.providers = List.copyOf(providers);
        this.templateEngine = templateEngine;
        this.redis = redisProvider.getIfAvailable();
        this.llmGenerator = llmGenerator;
        this.llmEnabled = llmProps.enabled();
    }

    public AiInsightResponse getInsight(long userId) {
        String cacheKey = cacheKey(userId);
        AiInsightResponse cached = readCache(cacheKey);
        if (cached != null) {
            return cached;
        }
        return recomputeAndCache(userId);
    }

    public AiInsightResponse refreshInsight(long userId) {
        return recomputeAndCache(userId);
    }

    private AiInsightResponse recomputeAndCache(long userId) {
        String cacheKey = cacheKey(userId);
        AiCollectContext ctx = AiCollectContext.nowInShanghai();
        List<MetricValue> collected = new ArrayList<>(providers.size());
        int successCount = 0;
        for (AiInsightProvider provider : providers) {
            if (!provider.isEnabled(userId)) {
                collected.add(MetricValue.none());
                continue;
            }
            try {
                MetricValue mv = provider.collect(userId, ctx);
                collected.add(mv);
                if (mv.isNonEmpty()) {
                    successCount++;
                }
            } catch (RuntimeException ex) {
                log.warn("AI provider failed: key={}, userId={}, err={}",
                    provider.key(), userId, ex.toString());
                collected.add(MetricValue.none());
            }
        }

        if (successCount == 0) {
            log.error("AI insight all providers failed: userId={}", userId);
            throw new BusinessException(ErrorCode.AI_DEGRADED,
                "AI 洞察数据暂时不可用，请稍后重试");
        }

        AiInsightResponse response = buildResponse(userId, collected, ctx.today());
        writeCache(cacheKey, response);
        return response;
    }

    /** v2.1 重构：先 LLM，失败回退模板；source 字段决定降级层级。 */
    private AiInsightResponse buildResponse(long userId, List<MetricValue> collected, LocalDate today) {
        MetricValue task = valueAt(collected, AiConstants.PROVIDER_TASK);
        MetricValue expense = valueAt(collected, AiConstants.PROVIDER_EXPENSE);
        MetricValue plan = valueAt(collected, AiConstants.PROVIDER_PLAN);

        // 默认：模板渲染
        String headline = renderHeadline(task, expense);
        String source = "template";
        String advice = null;
        String highlight = null;
        Mood mood = null;
        LlmMeta llmMeta = null;

        // L1：尝试 LLM（CLAUDE.md §11.3 L1 失败不重试）
        if (llmEnabled) {
            try {
                LlmInsightPayload payload = llmGenerator.generate(userId, collected, today);
                headline = payload.headline();
                advice = payload.advice();
                highlight = payload.highlight();
                mood = payload.mood();
                llmMeta = payload.llmMeta();
                source = "llm";
            } catch (RuntimeException ex) {
                log.warn("AI LLM fallback to template: userId={}, err={}", userId, ex.toString());
                // L2 已默认；source 保持 "template"
            }
        }

        List<AiChipDto> chips = List.of(
            chipForTask(task),
            chipForExpense(expense),
            chipForPlan(plan)
        );

        return new AiInsightResponse(
            headline,
            chips,
            Instant.now(),
            0L,
            source,
            advice,
            highlight,
            mood,
            llmMeta
        );
    }

    // ===== 私有辅助（chip 渲染、renderHeadline、cache 等）保持 v2.0 不变 =====

    private MetricValue valueAt(List<MetricValue> collected, String key) { /* 同 v2.0 */ }

    private String renderHeadline(MetricValue task, MetricValue expense) { /* 同 v2.0 */ }

    private AiChipDto chipForTask(MetricValue task) { /* 同 v2.0 */ }
    private AiChipDto chipForExpense(MetricValue expense) { /* 同 v2.0 */ }
    private AiChipDto chipForPlan(MetricValue plan) { /* 同 v2.0 */ }

    private static String trendName(MetricValue mv) { /* 同 v2.0 */ }
    private static String formatNumber(BigDecimal v, int scale) { /* 同 v2.0 */ }

    private String cacheKey(long userId) {
        return AiConstants.CACHE_KEY_PREFIX + userId;
    }

    private AiInsightResponse readCache(String key) { /* 同 v2.0 */ }

    /** v2.1：TTL 30min → 6h。 */
    private void writeCache(String key, AiInsightResponse response) {
        try {
            String json = OM.writeValueAsString(response);
            redis.opsForValue().set(key, json, CACHE_TTL_SECONDS, TimeUnit.SECONDS);
        } catch (RuntimeException | JsonProcessingException ex) {
            log.warn("AI cache write failed: key={}, err={}", key, ex.toString());
        }
    }

    public static long freshnessSeconds(AiInsightResponse r) { /* 同 v2.0 */ }
}
```

```typescript
import { defineStore } from 'pinia';
import { ref } from 'vue';
import type { AiInsight } from '@/types/ai';
import { aiApi } from '@/api/ai';

export const useAiInsightStore = defineStore('aiInsight', () => {
  const insight = ref<AiInsight | null>(null);
  const loading = ref(false);
  const error = ref<Error | null>(null);

  async function loadToday() {
    loading.value = true;
    error.value = null;
    try {
      insight.value = await aiApi.getToday();
    } catch (e) {
      error.value = e as Error;
      insight.value = null;
    } finally {
      loading.value = false;
    }
  }

  async function loadAnalysis() {
    // 共享缓存：若已有数据且新鲜（< 6h），直接复用；否则 fetch
    if (insight.value && insight.value.freshnessSeconds < 6 * 3600) {
      return insight.value;
    }
    loading.value = true;
    try {
      insight.value = await aiApi.getAnalysis();
      return insight.value;
    } finally {
      loading.value = false;
    }
  }

  async function refresh() {
    loading.value = true;
    try {
      insight.value = await aiApi.refresh();
    } finally {
      loading.value = false;
    }
  }

  return { insight, loading, error, loadToday, loadAnalysis, refresh };
});
```
