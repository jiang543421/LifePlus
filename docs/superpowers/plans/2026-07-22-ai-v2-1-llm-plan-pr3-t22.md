```vue
<template>
  <el-card class="ai-insight-card" @click="openDrawer">
    <template #header>
      <div class="header-row">
        <span class="title">✦ 今日洞察</span>
        <div class="actions">
          <el-tag v-if="insight?.source === 'llm'" type="success" size="small" effect="plain">AI 生成</el-tag>
          <el-tag v-else-if="insight?.source === 'template'" type="info" size="small" effect="plain">模板</el-tag>
          <el-button text size="small" :loading="loading" @click.stop="onRefresh">
            <el-icon><Refresh /></el-icon>
          </el-button>
          <el-button text size="small" @click.stop="openDrawer">
            <el-icon><InfoFilled /></el-icon>
          </el-button>
        </div>
      </div>
    </template>

    <div v-if="loading" class="loading-state">
      <el-skeleton :rows="2" animated />
    </div>
    <div v-else>
      <p class="headline">{{ insight?.headline || '加载中…' }}</p>
      <el-row :gutter="16" class="chip-row">
        <el-col :xs="24" :sm="8" v-for="chip in insight?.chips || []" :key="chip.key">
          <div class="chip">
            <div class="chip-label">{{ chip.label }}</div>
            <div class="chip-value">{{ chip.value }} {{ chip.unit }}</div>
            <div class="chip-delta" :class="trendClass(chip.trend)">{{ chip.deltaText }}</div>
          </div>
        </el-col>
      </el-row>
      <div class="footer-time">
        生成于 {{ formatTime(insight?.generatedAt) }}
        <span v-if="insight?.freshnessSeconds">· 缓存 {{ formatAge(insight.freshnessSeconds) }} 前</span>
      </div>
    </div>
  </el-card>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import { Refresh, InfoFilled } from '@element-plus/icons-vue';
import { useAiInsightStore } from '@/stores/aiInsight';
import { useRouter } from 'vue-router';

const store = useAiInsightStore();
const router = useRouter();
const insight = computed(() => store.insight);
const loading = computed(() => store.loading);

async function onRefresh() {
  await store.refresh();
}

function openDrawer() {
  router.push({ name: 'home' }).then(() => {
    // drawer 由父组件控制可见性；触发 store.loadToday() 同步
    store.loadToday();
    emit('open-drawer');
  });
}

const emit = defineEmits<{ 'open-drawer': [] }>();

function trendClass(t: string): string {
  return { UP: 'trend-up', DOWN: 'trend-down', FLAT: 'trend-flat', NONE: 'trend-none' }[t] || '';
}

function formatTime(iso?: string): string {
  if (!iso) return '';
  return new Date(iso).toLocaleString('zh-CN', { hour: '2-digit', minute: '2-digit' });
}

function formatAge(secs: number): string {
  const h = Math.floor(secs / 3600);
  const m = Math.floor((secs % 3600) / 60);
  return h > 0 ? `${h}h${m}m` : `${m}m`;
}
</script>

.footer-time { font-size: 12px; color: var(--el-text-color-secondary); margin-top: 8px; }
</style>
```
