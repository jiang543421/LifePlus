- [ ] **Step 2: 创建 `ai-insight.spec.ts`**

```typescript
import { test, expect } from '@playwright/test'

/**
 * AI 洞察 E2E 流程（spec §10.1）。
 *
 * 前置：本地 backend + frontend 已启动（docker compose up）。
 * 种子账号：test@example.com / Password123
 */

test.describe('AI Insight', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/login')
    await page.fill('input[name="email"]', 'test@example.com')
    await page.fill('input[name="password"]', 'Password123')
    await page.click('button[type="submit"]')
    await page.waitForURL('/')
  })

  test('首页加载 AI 卡片', async ({ page }) => {
    const card = page.locator('[data-test="card-root"]')
    await expect(card).toBeVisible({ timeout: 10_000 })
    await expect(page.locator('[data-test="headline"]')).toBeVisible({
      timeout: 15_000,
    })
  })

  test('点击卡片打开 drawer', async ({ page }) => {
    await page.locator('[data-test="card-root"]').waitFor({ timeout: 10_000 })
    await page.locator('[data-test="card-root"]').click()
    await expect(page.locator('[data-test="drawer"]')).toBeVisible()
    await expect(page.locator('[data-test="chip-taskCompletion"]')).toBeVisible({
      timeout: 5_000,
    })
  })

  test('刷新按钮触发 refresh 接口', async ({ page }) => {
    let refreshCount = 0
    page.on('request', (req) => {
      if (req.url().includes('/api/v1/ai/insight/refresh')) refreshCount++
    })

    await page.locator('[data-test="card-root"]').waitFor()
    await page.locator('[data-test="refresh-btn"]').click()

    await expect.poll(() => refreshCount, { timeout: 5_000 }).toBeGreaterThan(0)
  })

  test('新用户零数据：headline 为空态文案', async ({ page }) => {
    await page.goto('/register')
    const ts = Date.now()
    const email = `new-${ts}@example.com`
    await page.fill('input[name="email"]', email)
    await page.fill('input[name="password"]', 'Password123')
    await page.click('button[type="submit"]')
    await page.waitForURL('/')

    const card = page.locator('[data-test="card-root"]')
    await expect(card).toBeVisible({ timeout: 10_000 })

    const headline = page.locator('[data-test="headline"]')
    await expect(headline).toBeVisible({ timeout: 15_000 })
    await expect(headline).toContainText(/还没有|暂无|今日|/)
  })

  test('未登录访问 / 重定向到 login', async ({ page }) => {
    await page.context().clearCookies()
    await page.goto('/')
    expect(page.url()).toContain('/login')
  })
})
```
