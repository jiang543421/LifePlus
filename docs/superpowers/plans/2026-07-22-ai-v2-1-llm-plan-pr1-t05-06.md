- [ ] **Step 4: Create `.gitleaks.toml`**

```toml
title = "LifePulse secret scan"

[[rules]]
id = "deepseek-api-key"
description = "DeepSeek API Key"
regex = '''sk-[a-zA-Z0-9]{20,}'''
tags = ["secret", "api-key"]

[[rules]]
id = "openai-api-key"
description = "OpenAI / OpenAI-compatible API Key"
regex = '''sk-(proj-)?[a-zA-Z0-9]{20,}'''
tags = ["secret", "api-key"]

[[rules]]
id = "lp-llm-api-key-non-placeholder"
description = "LP_LLM_API_KEY with non-placeholder value"
regex = '''LP_LLM_API_KEY=(?!sk-replace-|.*\$\{)[a-zA-Z0-9-]{20,}'''
tags = ["secret", "env"]

[allowlist]
paths = [
  '''.env.example$''',
  '''docs/.*\.md$''',
  '''backend/src/test/.*\.java$''',
]
```

Run (with real key in `.env`): `cd backend && ./mvnw spring-boot:run`
