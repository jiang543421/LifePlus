package com.lifepulse.daily.web;

import com.lifepulse.common.exception.BusinessException;
import com.lifepulse.common.exception.ErrorCode;
import com.lifepulse.common.web.MyResponse;
import com.lifepulse.daily.DailyReportPayload;
import com.lifepulse.daily.WeeklyReportPayload;
import com.lifepulse.daily.service.DailyReportService;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.web.bind.annotation.*;

import java.time.LocalDate;
import java.util.regex.Pattern;

@RestController
@RequestMapping("/api/daily")
public class DailyReportController {

  private static final Pattern ISO_WEEK = Pattern.compile("^\\d{4}-W\\d{2}$");

  private final DailyReportService service;

  public DailyReportController(DailyReportService service) {
    this.service = service;
  }

  @GetMapping
  public MyResponse<DailyReportPayload> daily(
      @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate date) {
    LocalDate target = date == null ? LocalDate.now() : date;
    return MyResponse.ok(service.aggregateDaily(target));
  }

  @GetMapping("/week")
  public MyResponse<WeeklyReportPayload> weekly(
      @RequestParam(required = false) String week) {
    String target = week == null ? currentIsoWeek() : week;
    if (!ISO_WEEK.matcher(target).matches()) {
      throw new BusinessException(ErrorCode.VALIDATION, "周格式非法,应为 YYYY-Www");
    }
    return MyResponse.ok(service.aggregateWeekly(target));
  }

  private String currentIsoWeek() {
    LocalDate today = LocalDate.now();
    int year = today.get(java.time.temporal.ChronoField.YEAR_OF_WEEK_BASED_YEAR);
    int w = today.get(java.time.temporal.ChronoField.WEEK_OF_WEEK_BASED_YEAR);
    return String.format("%d-W%02d", year, w);
  }
}

package com.lifepulse.daily.web;

import com.lifepulse.common.exception.BusinessException;
import com.lifepulse.common.exception.ErrorCode;
import com.lifepulse.common.web.GlobalExceptionHandler;
import com.lifepulse.daily.DailyReportPayload;
import com.lifepulse.daily.WeeklyReportPayload;
import com.lifepulse.daily.service.DailyReportService;
import com.lifepulse.daily.*;
import com.lifepulse.security.JwtAuthFilter;
import com.lifepulse.security.UserContext;
import org.junit.jupiter.api.*;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.boot.test.mock.mockito.MockBean;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Import;
import org.springframework.test.web.servlet.MockMvc;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.List;
import java.util.Map;

import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.*;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;

@WebMvcTest(DailyReportController.class)
@Import({GlobalExceptionHandler.class, DailyReportControllerWebTest.StubFilterConfig.class})
class DailyReportControllerWebTest {

  @Autowired MockMvc mvc;
  @MockBean DailyReportService service;

  @org.springframework.boot.test.context.TestConfiguration
  static class StubFilterConfig {
    @Bean JwtAuthFilter jwtAuthFilter() {
      return new JwtAuthFilter() {
        @Override protected void doFilterInternal(jakarta.servlet.http.HttpServletRequest req,
                                                    jakarta.servlet.http.HttpServletResponse resp,
                                                    jakarta.servlet.FilterChain chain)
            throws java.io.IOException, jakarta.servlet.ServletException {
          if (req.getHeader("X-Test-User") != null) {
            UserContext.set(Long.parseLong(req.getHeader("X-Test-User")));
          }
          chain.doFilter(req, resp);
          UserContext.clear();
        }
      };
    }
  }

  private static final String AUTH = "X-Test-User: 1";

  private DailyReportPayload mkPayload() {
    return new DailyReportPayload(
        LocalDate.of(2026, 7, 15),
        new TaskMetrics(3, 5, 0.6, Map.of(), Map.of()),
        new PlanMetrics(2, 240, Map.of(), 14),
        new ExpenseMetrics(new BigDecimal("47.00"), 2, Map.of(), List.of()),
        new DietMetrics(false, null, "饮食模块未上线"));
  }

  @Test void daily_unauthorized_returns401() throws Exception {
    mvc.perform(get("/api/daily")).andExpect(status().isUnauthorized());
  }

  @Test void daily_happy_returns200() throws Exception {
    when(service.aggregateDaily(any())).thenReturn(mkPayload());
    mvc.perform(get("/api/daily?date=2026-07-15").header("X-Test-User", "1"))
        .andExpect(status().isOk())
        .andExpect(jsonPath("$.data.task.completionRate").value(0.6))
        .andExpect(jsonPath("$.data.diet.enabled").value(false));
  }

  @Test void daily_futureDate_returns1004() throws Exception {
    when(service.aggregateDaily(any()))
        .thenThrow(new BusinessException(ErrorCode.VALIDATION, "不支持未来日期"));
    mvc.perform(get("/api/daily?date=2099-01-01").header("X-Test-User", "1"))
        .andExpect(status().isOk())
        .andExpect(jsonPath("$.code").value(1004));
  }

  @Test void daily_invalidDateFormat_returns400() throws Exception {
    mvc.perform(get("/api/daily?date=not-a-date").header("X-Test-User", "1"))
        .andExpect(status().isBadRequest());
  }

  @Test void week_happy_returns200() throws Exception {
    when(service.aggregateWeekly("2026-W29")).thenReturn(
        new WeeklyReportPayload("2026-W29", LocalDate.of(2026, 7, 13),
            LocalDate.of(2026, 7, 19), null));
    mvc.perform(get("/api/daily/week?week=2026-W29").header("X-Test-User", "1"))
        .andExpect(status().isOk())
        .andExpect(jsonPath("$.data.isoWeek").value("2026-W29"));
  }

  @Test void week_invalidFormat_returns1004() throws Exception {
    mvc.perform(get("/api/daily/week?week=2026-29").header("X-Test-User", "1"))
        .andExpect(status().isOk())
        .andExpect(jsonPath("$.code").value(1004));
  }

  @Test void week_missingParam_defaultsToCurrent() throws Exception {
    when(service.aggregateWeekly(anyString())).thenReturn(
        new WeeklyReportPayload("2026-W29", LocalDate.of(2026, 7, 13),
            LocalDate.of(2026, 7, 19), null));
    mvc.perform(get("/api/daily/week").header("X-Test-User", "1"))
        .andExpect(status().isOk());
  }
}
